45  CV_Assert(!descriptors.empty());
46  int descCount = 0;
47  for(size_t i = 0; i < descriptors.size(); i++)
48  descCount += descriptors[i].rows;

50  Mat mergedDescriptors(descCount, descriptors[0].cols,
