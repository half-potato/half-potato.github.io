56 std::vector<TrackableKFStruct, Eigen::aligned_allocator<TrackableKFStruct> > TrackableKeyFrameSearch::findEuclideanOverlapFrames(Frame* frame, float distanceTH, float angleTH, bool checkBothScales)
57 {
58  // basically the maximal angle-difference in viewing direction is angleTH*(average FoV).
59  // e.g. if the FoV is 130°, then it is angleTH*130°.
60  float cosAngleTH = cosf(angleTH*0.5f*(fowX + fowY));
61 
62 
63  Eigen::Vector3d pos = frame->getScaledCamToWorld().translation();
64  Eigen::Vector3d viewingDir = frame->getScaledCamToWorld().rotationMatrix().rightCols<1>();
65 
66  std::vector<TrackableKFStruct, Eigen::aligned_allocator<TrackableKFStruct> > potentialReferenceFrames;
67 
68  float distFacReciprocal = 1;
69  if(checkBothScales)
70  distFacReciprocal = frame->meanIdepth / frame->getScaledCamToWorld().scale();
71 
72  // for each frame, calculate the rough score, consisting of pose, scale and angle overlap.
73  graph->keyframesAllMutex.lock_shared();
74  for(unsigned int i=0;i<graph->keyframesAll.size();i++)
75  {
76  Eigen::Vector3d otherPos = graph->keyframesAll[i]->getScaledCamToWorld().translation();
77 
78  // get distance between the frames, scaled to fit the potential reference frame.
79  float distFac = graph->keyframesAll[i]->meanIdepth / graph->keyframesAll[i]->getScaledCamToWorld().scale();
80  if(checkBothScales && distFacReciprocal < distFac) distFac = distFacReciprocal;
81  Eigen::Vector3d dist = (pos - otherPos) * distFac;
82  float dNorm2 = dist.dot(dist);
83  if(dNorm2 > distanceTH) continue;
84 
85  Eigen::Vector3d otherViewingDir = graph->keyframesAll[i]->getScaledCamToWorld().rotationMatrix().rightCols<1>();
86  float dirDotProd = otherViewingDir.dot(viewingDir);
87  if(dirDotProd < cosAngleTH) continue;
88 
89  potentialReferenceFrames.push_back(TrackableKFStruct());
90  potentialReferenceFrames.back().ref = graph->keyframesAll[i];
91  potentialReferenceFrames.back().refToFrame = se3FromSim3(graph->keyframesAll[i]->getScaledCamToWorld().inverse() * frame->getScaledCamToWorld()).inverse();
92  potentialReferenceFrames.back().dist = dNorm2;
93  potentialReferenceFrames.back().angle = dirDotProd;
94  }
95  graph->keyframesAllMutex.unlock_shared();
96 
97  return potentialReferenceFrames;
98 }
