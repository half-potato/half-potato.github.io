61 void BagLoader::loadBag(std::string filename)
62 {
63  ScopedTimer s(__FUNCTION__);
64  ros::NodeHandle nh;
65  ParameterServer* ps = ParameterServer::instance();
66  std::string points_tpc = ps->get<std::string>("individual_cloud_out_topic").c_str();//ps->get<std::string>("topic_points");
67  ROS_INFO_STREAM("Listening to " << points_tpc);
68  std::string tf_tpc = std::string("/tf");
69  tf_pub_ = nh.advertise<tf::tfMessage>(tf_tpc, 10);
70 
71  ROS_INFO("Loading Bagfile %s", filename.c_str());
72  Q_EMIT setGUIStatus(QString("Loading ")+filename.c_str());
73  { //bag will be destructed after this block (hopefully frees memory for the optimizer)
74  rosbag::Bag bag;
75  try{
76  bag.open(filename, rosbag::bagmode::Read);
77  } catch(rosbag::BagIOException ex) {
78  ROS_FATAL("Opening Bagfile %s failed: %s Quitting!", filename.c_str(), ex.what());
79  ros::shutdown();
80  return;
81  }
82  ROS_INFO("Opened Bagfile %s", filename.c_str());
83  Q_EMIT setGUIStatus(QString("Opened ")+filename.c_str());
84 
85  // Image topics to load for bagfiles
86  std::vector<std::string> topics;
87  topics.push_back(points_tpc);
88  topics.push_back(tf_tpc);
89 
90  rosbag::View view(bag, rosbag::TopicQuery(topics));
91  // Simulate sending of the messages in the bagfile
92  std::deque<sensor_msgs::PointCloud2::ConstPtr> pointclouds;
93  ros::Time last_tf(0);
94  BOOST_FOREACH(rosbag::MessageInstance const m, view)
95  {
96  if(!ros::ok()) return;
97  while(pause_) {
98  usleep(100);
99  ROS_INFO_THROTTLE(5.0,"Paused - press Space to unpause");
100  if(!ros::ok()) return;
101  }
102 
103  ROS_INFO_STREAM("Processing message on topic " << m.getTopic());
104  if (m.getTopic() == points_tpc || ("/" + m.getTopic() == points_tpc))
105  {
106  sensor_msgs::PointCloud2::ConstPtr pointcloud = m.instantiate<sensor_msgs::PointCloud2>();
107  //if (cam_info) cam_info_sub_->newMessage(cam_info);
108  if (pointcloud) pointclouds.push_back(pointcloud);
109  ROS_INFO("Found Message of %s", points_tpc.c_str());
110  }
111  if (m.getTopic() == tf_tpc|| ("/" + m.getTopic() == tf_tpc)){
112  tf::tfMessage::ConstPtr tf_msg = m.instantiate<tf::tfMessage>();
113  if (tf_msg) {
114  //if(tf_msg->transforms[0].header.frame_id == "/kinect") continue;//avoid destroying tf tree if odom is used
115  //prevents missing callerid warning
116  boost::shared_ptr<std::map<std::string, std::string> > msg_header_map = tf_msg->__connection_header;
117  (*msg_header_map)["callerid"] = "rgbdslam";
118  tf_pub_.publish(tf_msg);
119  ROS_INFO("Found Message of %s", tf_tpc.c_str());
120  last_tf = tf_msg->transforms[0].header.stamp;
121  last_tf -= ros::Duration(0.3);
122  }
123  }
124  while(!pointclouds.empty() && pointclouds.front()->header.stamp < last_tf){
125  Q_EMIT setGUIInfo(QString("Processing frame ") + QString::number(data_id_));
126  callback(pointclouds.front());
127  pointclouds.pop_front();
128  }
129 
130  }
131  bag.close();
132  }
133  ROS_INFO("Bagfile fully processed");
134  Q_EMIT setGUIStatus(QString("Done with ")+filename.c_str());
135  Q_EMIT bagFinished();
136 }

139 void BagLoader::callback(const sensor_msgs::PointCloud2ConstPtr& point_cloud)
140 {
141  ScopedTimer s(__FUNCTION__);
142  ROS_INFO("Processing cloud");
143  if(++data_id_ < ParameterServer::instance()->get<int>("skip_first_n_frames")
144  || data_id_ % ParameterServer::instance()->get<int>("data_skip_step") != 0){
145  ROS_INFO_THROTTLE(1, "Skipping Frame %i because of data_skip_step setting (this msg is only shown once a sec)", data_id_);
146  return;
147  };
148 
149  pointcloud_type* pc_col = new pointcloud_type();//will belong to viewer?
150  pcl::fromROSMsg(*point_cloud,*pc_col);
151  sendWithTransformation(pc_col);
152 }

168 void BagLoader::sendWithTransformation(pointcloud_type* cloud)
169 {
170  ScopedTimer s(__FUNCTION__);
171  ParameterServer* ps = ParameterServer::instance();
172  std::string fixed_frame = ps->get<std::string>("fixed_frame_name");
173  std::string depth_frame_id = cloud->header.frame_id;
174  ros::Time depth_time = cloud->header.stamp;
175  tf::StampedTransform map2points;
176 
177  try{
178  tflistener_->waitForTransform(fixed_frame, depth_frame_id, depth_time, ros::Duration(0.005));
179  tflistener_->lookupTransform( fixed_frame, depth_frame_id, depth_time,map2points);
180  //printTransform("Current Transform", map2points);
181 
182  QMatrix4x4 transform_to_map = g2o2QMatrix(tf2G2O(map2points));
183  Q_EMIT setPointCloud(cloud, transform_to_map);
184  }
185  catch (tf::TransformException ex){
186  ROS_ERROR("%s - No transformation available",ex.what());
187  }
188 }
