201 void ParameterServer::getValues() {
202  std::map<std::string, boost::any>::const_iterator itr;
203  for (itr = config.begin(); itr != config.end(); ++itr) {
204  std::string name = itr->first;
205  if (itr->second.type() == typeid(std::string)) {
206  config[name] = getFromParameterServer<std::string> (pre + name,
207  boost::any_cast<std::string>(itr->second));
208  ROS_DEBUG_STREAM("Value for " << name << ": " << boost::any_cast<std::string>(itr->second));
209  } else if (itr->second.type() == typeid(int)) {
210  config[name] = getFromParameterServer<int> (pre + name,
211  boost::any_cast<int>(itr->second));
212  ROS_DEBUG_STREAM("Value for " << name << ": " << boost::any_cast<int>(itr->second));
213  } else if (itr->second.type() == typeid(double)) {
214  config[name] = getFromParameterServer<double> (pre + name,
215  boost::any_cast<double>(itr->second));
216  ROS_DEBUG_STREAM("Value for " << name << ": " << boost::any_cast<double>(itr->second));
217  } else if (itr->second.type() == typeid(bool)) {
218  config[name] = getFromParameterServer<bool> (pre + name,
219  boost::any_cast<bool>(itr->second));
220  ROS_DEBUG_STREAM("Value for " << name << ": " << boost::any_cast<bool>(itr->second));
221  }
222  }
223  checkValues();
224 }

226 void ParameterServer::checkValues() {
227  if (get<std::string>("matcher_type").compare("SIFTGPU") == 0
228  && get<bool>("concurrent_node_construction") == true) {
229  config["concurrent_node_construction"] = static_cast<bool>(false);
230  ROS_WARN("Cannot use concurrent node construction with SiftGPU matcher! 'concurrent_node_construction' was set to false. Everything should work fine, but the CPU-threading won't happen (because you are using the GPU instead).");
231  }
232 
233  if (get<double>("voxelfilter_size") > 0 && get<double>("observability_threshold") > 0) {
234  ROS_ERROR("You cannot use the voxelfilter (param: voxelfilter_size) in combination with the environment measurement model (param: observability_threshold)");
235  }
236  if (get<std::string>("matcher_type").compare("SIFTGPU") == 0
237  && get<bool>("concurrent_edge_construction") == true) {
238  config["concurrent_edge_construction"] = static_cast<bool>(false);
239  ROS_WARN("Cannot use concurrent edge construction with SiftGPU matcher! 'concurrent_edge_construction' was set to false. Everything should work fine, but the CPU-threading won't happen (because you are using the GPU instead).");
240  }
241  if (get<double>("max_translation_meter") <= 0){
242  double dInf = std::numeric_limits<double>::infinity();
243  set("max_translation_meter", dInf);
244  }
245  if (get<double>("max_rotation_degree") <= 0){
246  double dInf = std::numeric_limits<double>::infinity();
247  set("max_rotation_degree", dInf);
248  }
249 }
