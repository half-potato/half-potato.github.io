204 QList<int> GraphManager::getPotentialEdgeTargetsWithDijkstra(const Node* new_node, int sequential_targets, int geodesic_targets, int sampled_targets, int predecessor_id, bool include_predecessor)
205 {
206  QList<int> ids_to_link_to; //return value
207  if(predecessor_id < 0) predecessor_id = graph_.size()-1;
208  //Prepare output
209  std::stringstream ss;
210  ss << "Node ID's to compare with candidate for node " << graph_.size() << ". Sequential: ";
211 
212  if((int)camera_vertices.size() <= sequential_targets+geodesic_targets+sampled_targets ||
213  camera_vertices.size() <= 1)
214  { //if less prev nodes available than targets requestet, just use all
215  sequential_targets = sequential_targets+geodesic_targets+sampled_targets;
216  geodesic_targets = 0;
217  sampled_targets = 0;
218  predecessor_id = graph_.size()-1;
219  }
220 
221  if(sequential_targets > 0){
222  //all the sequential targets (will be checked last)
223  for (int i=1; i < sequential_targets+1 && predecessor_id-i >= 0; i++) {
224  ids_to_link_to.push_back(predecessor_id-i);
225  ss << ids_to_link_to.back() << ", " ;
226  }
227  }
228 
229  if(geodesic_targets > 0){
230  g2o::HyperDijkstra hypdij(optimizer_);
231  g2o::UniformCostFunction cost_function;
232  g2o::VertexSE3* prev_vertex = dynamic_cast<g2o::VertexSE3*>(optimizer_->vertex(graph_[predecessor_id]->vertex_id_));
233  hypdij.shortestPaths(prev_vertex,&cost_function,ParameterServer::instance()->get<int>("geodesic_depth"));
234  g2o::HyperGraph::VertexSet& vs = hypdij.visited();
235 
236  //Need to convert vertex_id to node id
237  std::map<int, int> vertex_id_to_node_id;
238  for (graph_it it = graph_.begin(); it !=graph_.end(); ++it){
239  Node *node = it->second;
240  vertex_id_to_node_id[node->vertex_id_] = node->id_;
241  //ROS_WARN("ID Pair: %d, %d", node->vertex_id_, node->id_);
242  }
243 
244  //Geodesic Neighbours except sequential
245  std::map<int,int> neighbour_indices; //maps neighbour ids to their weights in sampling
246  int sum_of_weights=0;
247  for (g2o::HyperGraph::VertexSet::iterator vit=vs.begin(); vit!=vs.end(); vit++) { //FIXME: Mix of vertex id and graph node (with features) id
248  int vid = (*vit)->id();
249  //ROS_WARN("Vertex ID: %d", vid);
250  int id = 0;
251  try{
252  id = vertex_id_to_node_id.at(vid);
253  }
254  catch (std::exception e){//Catch exceptions: Unexpected problems shouldn't crash the application
255  ROS_ERROR("Vertex ID %d has no corresponding node", vid);
256  ROS_ERROR("Map Content:");
257  for(std::map<int,int>::const_iterator it = vertex_id_to_node_id.begin(); it != vertex_id_to_node_id.end(); it++){
258  ROS_ERROR("Node ID %d: Vertex ID %d", it->first, it->second);
259  }
260  for(g2o::SparseOptimizer::VertexIDMap::iterator it = optimizer_->vertices().begin(); it != optimizer_->vertices().end(); it++)
261  {
262  ROS_ERROR("Vertex ID %d", it->first);
263  }
264  }
265  if(!graph_.at(id)->matchable_) continue;
266  if(id < predecessor_id-sequential_targets || (id > predecessor_id && id <= (int)graph_.size()-1)){ //Geodesic Neighbours except sequential
267  int weight = abs(predecessor_id-id);
268  neighbour_indices[id] = weight; //higher probability to be drawn if far away
269  sum_of_weights += weight;
270  }
271  }
272 
273  //Sample targets from graph-neighbours
274  ss << "Dijkstra: ";
275  while(ids_to_link_to.size() < sequential_targets+geodesic_targets && neighbour_indices.size() != 0){
276  int random_pick = rand() % sum_of_weights;
277  ROS_DEBUG("Pick: %d/%d", random_pick, sum_of_weights);
278  int weight_so_far = 0;
279  for(std::map<int,int>::iterator map_it = neighbour_indices.begin(); map_it != neighbour_indices.end(); map_it++ ){
280  weight_so_far += map_it->second;
281  ROS_DEBUG("Checking: %d, %d, %d", map_it->first, map_it-> second, weight_so_far);
282  if(weight_so_far > random_pick){//found the selected one
283  int sampled_id = map_it->first;
284  ids_to_link_to.push_front(sampled_id);
285  ss << ids_to_link_to.front() << ", " ;
286  sum_of_weights -= map_it->second;
287  ROS_DEBUG("Taking ID: %d, decreasing sum of weights to %d", map_it->first, sum_of_weights);
288  neighbour_indices.erase(map_it);
289  ROS_ERROR_COND(sum_of_weights<0, "Sum of weights should never be zero");
290  break;
291  }
292  ROS_DEBUG("Skipping ID: %d", map_it->first);
293  }//for
294  }
295  }
296 
297  if(sampled_targets > 0){
298  ss << "Random Sampling: ";
299  std::vector<int> non_neighbour_indices;//initially holds all, then neighbours are deleted
300  non_neighbour_indices.reserve(graph_.size());
301  for (QList<int>::iterator it = keyframe_ids_.begin(); it != keyframe_ids_.end(); it++){
302  if(ids_to_link_to.contains(*it) == 0 && graph_.at(*it)->matchable_){
303  non_neighbour_indices.push_back(*it);
304  }
305  }
306 
307  //Sample targets from non-neighbours (search new loops)
308  while(ids_to_link_to.size() < geodesic_targets+sampled_targets+sequential_targets && non_neighbour_indices.size() != 0){
309  int index_of_v_id = rand() % non_neighbour_indices.size();
310  int sampled_id = non_neighbour_indices[index_of_v_id];
311  non_neighbour_indices[index_of_v_id] = non_neighbour_indices.back(); //copy last id to position of the used id
312  non_neighbour_indices.resize(non_neighbour_indices.size()-1); //drop last id
313  ids_to_link_to.push_front(sampled_id);
314  ss << ids_to_link_to.front() << ", " ;
315  }
316  }
317 
318  if(include_predecessor){
319  ids_to_link_to.push_back(predecessor_id);
320  ss << predecessor_id;
321  }
322  ROS_INFO("%s", ss.str().c_str());
323  return ids_to_link_to; //only compare to first frame
324 }
