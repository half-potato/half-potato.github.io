121  void GICPPointSet::ComputeMatrices() {
122  pthread_mutex_lock(&mutex_);
123  if(kdtree_ == NULL) {
124  return;
125  }
126  if(matrices_done_) {
127  return;
128  }
129  matrices_done_ = true;
130  pthread_mutex_unlock(&mutex_);
131 
132  int N = NumPoints();
133  int K = 20; // number of closest points to use for local covariance estimate
134  double mean[3];
135 
136  ANNpoint query_point = annAllocPt(3);
137 
138  ANNdist *nn_dist_sq = new ANNdist[K];
139  if(nn_dist_sq == NULL) {
140  //TODO: handle this
141  }
142  ANNidx *nn_indecies = new ANNidx[K];
143  if(nn_indecies == NULL) {
144  //TODO: handle this
145  }
146  gsl_vector *work = gsl_vector_alloc(3);
147  if(work == NULL) {
148  //TODO: handle
149  }
150  gsl_vector *gsl_singulars = gsl_vector_alloc(3);
151  if(gsl_singulars == NULL) {
152  //TODO: handle
153  }
154  gsl_matrix *gsl_v_mat = gsl_matrix_alloc(3, 3);
155  if(gsl_v_mat == NULL) {
156  //TODO: handle
157  }
158 
159  for(int i = 0; i < N; i++) {
160  query_point[0] = point_[i].x;
161  query_point[1] = point_[i].y;
162  query_point[2] = point_[i].z;
163 
164  gicp_mat_t &cov = point_[i].C;
165  // zero out the cov and mean
166  for(int k = 0; k < 3; k++) {
167  mean[k] = 0.;
168  for(int l = 0; l < 3; l++) {
169  cov[k][l] = 0.;
170  }
171  }
172 
173  kdtree_->annkSearch(query_point, K, nn_indecies, nn_dist_sq, 0.0);
174 
175  // find the covariance matrix
176  for(int j = 0; j < K; j++) {
177  GICPPoint &pt = point_[nn_indecies[j]];
178 
179  mean[0] += pt.x;
180  mean[1] += pt.y;
181  mean[2] += pt.z;
182 
183  cov[0][0] += pt.x*pt.x;
184 
185  cov[1][0] += pt.y*pt.x;
186  cov[1][1] += pt.y*pt.y;
187 
188  cov[2][0] += pt.z*pt.x;
189  cov[2][1] += pt.z*pt.y;
190  cov[2][2] += pt.z*pt.z;
191  }
192 
193  mean[0] /= (double)K;
194  mean[1] /= (double)K;
195  mean[2] /= (double)K;
196  // get the actual covariance
197  for(int k = 0; k < 3; k++) {
198  for(int l = 0; l <= k; l++) {
199  cov[k][l] /= (double)K;
200  cov[k][l] -= mean[k]*mean[l];
201  cov[l][k] = cov[k][l];
202  }
203  }
204 
205  // compute the SVD
206  gsl_matrix_view gsl_cov = gsl_matrix_view_array(&cov[0][0], 3, 3);
207  gsl_linalg_SV_decomp(&gsl_cov.matrix, gsl_v_mat, gsl_singulars, work);
208 
209  // zero out the cov matrix, since we know U = V since C is symmetric
210  for(int k = 0; k < 3; k++) {
211  for(int l = 0; l < 3; l++) {
212  cov[k][l] = 0;
213  }
214  }
215 
216  // reconstitute the covariance matrix with modified singular values using the column vectors in V.
217  for(int k = 0; k < 3; k++) {
218  gsl_vector_view col = gsl_matrix_column(gsl_v_mat, k);
219 
220  double v = 1.; // biggest 2 singular values replaced by 1
221  if(k == 2) { // smallest singular value replaced by gicp_epsilon
222  v = gicp_epsilon_;
223  }
224 
225  gsl_blas_dger(v, &col.vector, &col.vector, &gsl_cov.matrix);
226  }
227  }
228 
229  if(nn_dist_sq != NULL) {
230  delete [] nn_dist_sq;
231  }
232  if(nn_indecies != NULL) {
233  delete [] nn_indecies;
234  }
235  if(work != NULL) {
236  gsl_vector_free(work);
237  }
238  if(gsl_v_mat != NULL) {
239  gsl_matrix_free(gsl_v_mat);
240  }
241  if(gsl_singulars != NULL) {
242  gsl_vector_free(gsl_singulars);
243  }
244  query_point;
245  }

247  int GICPPointSet::AlignScan(GICPPointSet *scan, dgc_transform_t base_t, dgc_transform_t t, double max_match_dist, bool save_error_plot)
248  {
249  double max_d_sq = pow(max_match_dist, 2);
250  int num_matches = 0;
251  int n = scan->NumPoints();
252  double delta = 0.;
253  dgc_transform_t t_last;
254  ofstream fout_corresp;
255  ANNdist nn_dist_sq;
256  ANNidx *nn_indecies = new ANNidx[n];
257  ANNpoint query_point = annAllocPt(3);
258 
259  if(nn_indecies == NULL) {
260  //TODO: fail here
261  }
262 
263  gicp_mat_t *mahalanobis = new gicp_mat_t[n];
264  if(mahalanobis == NULL) {
265  //TODO: fail here
266  }
267  gsl_matrix *gsl_R = gsl_matrix_alloc(3, 3);
268  if(gsl_R == NULL) {
269  //TODO: fail here
270  }
271  gsl_matrix *gsl_temp = gsl_matrix_alloc(3, 3);
272  if(gsl_temp == NULL) {
273  //TODO: fail here
274  }
275 
276  bool converged = false;
277  int iteration = 0;
278  bool opt_status = false;
279 
280 
281  /* set up the optimization parameters */
282  GICPOptData opt_data;
283  opt_data.nn_indecies = nn_indecies;
284  opt_data.p1 = scan;
285  opt_data.p2 = this;
286  opt_data.M = mahalanobis;
287  opt_data.solve_rotation = solve_rotation_;
288  dgc_transform_copy(opt_data.base_t, base_t);
289 
290  GICPOptimizer opt;
291  opt.SetDebug(debug_);
292  opt.SetMaxIterations(max_iteration_inner_);
293  /* set up the mahalanobis matricies */
294  /* these are identity for now to ease debugging */
295  for(int i = 0; i < n; i++) {
296  for(int k = 0; k < 3; k++) {
297  for(int l = 0; l < 3; l++) {
298  mahalanobis[i][k][l] = (k == l)?1:0.;
299  }
300  }
301  }
302 
303  if(debug_) {
304  dgc_transform_write(base_t, "t_base.tfm");
305  dgc_transform_write(t, "t_0.tfm");
306  }
307 
308  while(!converged) {
309  dgc_transform_t transform_R;
310  dgc_transform_copy(transform_R, base_t);
311  dgc_transform_left_multiply(transform_R, t);
312  // copy the rotation component of the current total transformation (including base), into a gsl matrix
313  for(int i = 0; i < 3; i++) {
314  for(int j = 0; j < 3; j++) {
315  gsl_matrix_set(gsl_R, i, j, transform_R[i][j]);
316  }
317  }
318  if(debug_) {
319  fout_corresp.open("correspondence.txt");
320  }
321  /* find correpondences */
322  num_matches = 0;
323  for (int i = 0; i < n; i++) {
324  query_point[0] = scan->point_[i].x;
325  query_point[1] = scan->point_[i].y;
326  query_point[2] = scan->point_[i].z;
327 
328  dgc_transform_point(&query_point[0], &query_point[1],
329  &query_point[2], base_t);
330  dgc_transform_point(&query_point[0], &query_point[1],
331  &query_point[2], t);
332 
333  kdtree_->annkSearch(query_point, 1, &nn_indecies[i], &nn_dist_sq, 0.0);
334 
335  if (nn_dist_sq < max_d_sq) {
336  if(debug_) {
337  fout_corresp << i << "\t" << nn_indecies[i] << endl;
338  }
339 
340  // set up the updated mahalanobis matrix here
341  gsl_matrix_view C1 = gsl_matrix_view_array(&scan->point_[i].C[0][0], 3, 3);
342  gsl_matrix_view C2 = gsl_matrix_view_array(&point_[nn_indecies[i]].C[0][0], 3, 3);
343  gsl_matrix_view M = gsl_matrix_view_array(&mahalanobis[i][0][0], 3, 3);
344  gsl_matrix_set_zero(&M.matrix);
345  gsl_matrix_set_zero(gsl_temp);
346 
347  // M = R*C1 // using M as a temp variable here
348  gsl_blas_dgemm(CblasNoTrans, CblasNoTrans, 1., gsl_R, &C1.matrix, 1., &M.matrix);
349 
350  // temp = M*R' // move the temp value to 'temp' here
351  gsl_blas_dgemm(CblasNoTrans, CblasTrans, 1., &M.matrix, gsl_R, 0., gsl_temp);
352 
353  // temp += C2
354  gsl_matrix_add(gsl_temp, &C2.matrix);
355  // at this point temp = C2 + R*C1*R'
356 
357  // now invert temp to get the mahalanobis distance metric for gicp
358  // M = temp^-1
359  gsl_matrix_set_identity(&M.matrix);
360  gsl_linalg_cholesky_decomp(gsl_temp);
361  for(int k = 0; k < 3; k++) {
362  auto gsl_temp2 = gsl_matrix_row(&M.matrix, k);
363  gsl_linalg_cholesky_svx(gsl_temp, &(gsl_temp2.vector));
364  }
365  num_matches++;
366  }
367  else {
368  nn_indecies[i] = -1; // no match
369  }
370  }
371 
372  if(debug_) { // save the current M matrices to file for debugging
373  ofstream out("mahalanobis.txt");
374  if(out) {
375  for(int i = 0; i < n; i++) {
376  for(int k = 0; k < 3; k++) {
377  for(int l = 0; l < 3; l++) {
378  out << mahalanobis[i][k][l] << "\t";
379  }
380  }
381  out << endl;
382  }
383  }
384  out.close();
385  }
386 
387  if(debug_) {
388  fout_corresp.close();
389  }
390  opt_data.num_matches = num_matches;
391 
392  /* optimize transformation using the current assignment and Mahalanobis metrics*/
393  dgc_transform_copy(t_last, t);
394  opt_status = opt.Optimize(t, opt_data);
395 
396  if(debug_) {
397  cout << "Optimizer converged in " << opt.Iterations() << " iterations." << endl;
398  cout << "Status: " << opt.Status() << endl;
399 
400  std::ostringstream filename;
401  filename << "t_" << iteration+1 << ".tfm";
402  dgc_transform_write(t, filename.str().c_str());
403  }
404 
405  /* compute the delta from this iteration */
406  delta = 0.;
407  for(int k = 0; k < 4; k++) {
408  for(int l = 0; l < 4; l++) {
409  double ratio = 1;
410  if(k < 3 && l < 3) { // rotation part of the transform
411  ratio = 1./epsilon_rot_;
412  }
413  else {
414  ratio = 1./epsilon_;
415  }
416  double c_delta = ratio*fabs(t_last[k][l] - t[k][l]);
417 
418  if(c_delta > delta) {
419  delta = c_delta;
420  }
421  }
422  }
423  if(debug_) {
424  cout << "delta = " << delta << endl;
425  }
426 
427  /* check convergence */
428  iteration++;
429  if(iteration >= max_iteration_ || delta < 1) {
430  converged = true;
431  }
432  }
433  if(debug_) {
434  cout << "Converged in " << iteration << " iterations." << endl;
435  if(save_error_plot) {
436  opt.PlotError(t, opt_data, "error_func");
437  }
438  }
439  if(nn_indecies != NULL) {
440  delete [] nn_indecies;
441  }
442  if(mahalanobis != NULL) {
443  delete [] mahalanobis;
444  }
445  if(gsl_R != NULL) {
446  gsl_matrix_free(gsl_R);
447  }
448  if(gsl_temp != NULL) {
449  gsl_matrix_free(gsl_temp);
450  }
451  annDeallocPt(query_point);
452 
453  return iteration;
