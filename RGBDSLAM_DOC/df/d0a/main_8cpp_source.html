110 int main(int argc, char** argv)
111 {
112  setlocale(LC_NUMERIC,"C");//Avoid expecting german decimal separators in launch files
113 
114  //create thread object, to run the ros event processing loop in parallel to the qt loop
115  QtROS qtRos(argc, argv, "rgbdslam"); //ros node name & namespace
116 
117  //Depending an use_gui on the Parameter Server, a gui- or a headless application is used
118  QApplication app(argc, argv, ParameterServer::instance()->get<bool>("use_gui"));
119 
120  GraphManager graph_mgr;
121  //Instantiate the kinect image listener
122  OpenNIListener listener(&graph_mgr);
123  std::string bagfile_name = ParameterServer::instance()->get<std::string>("bagfile_name");
124  if(!bagfile_name.empty())
125  {
126  QObject::connect(&listener, SIGNAL(bagFinished()), &app, SLOT(quit()));
127  QObject::connect(&listener, SIGNAL(bagFinished()), &qtRos, SLOT(quitNow()));
128  listener.loadBagFileFromGUI(bagfile_name.c_str());
129  //QtConcurrent::run(&listener, &OpenNIListener::loadBag, bagfile_name);
130  }
131 
132  Graphical_UI* gui = NULL;
133  if (app.type() == QApplication::GuiClient){
134  gui = new Graphical_UI();
135  gui->show();
136  gui_connections(gui, &graph_mgr, &listener);
137  ui_connections(gui, &graph_mgr, &listener);//common connections for the user interfaces
138  } else {
139  ROS_WARN("Running without graphical user interface! See README or wiki page for how to interact with RGBDSLAM.");
140  }
141  //Create Ros service interface with or without gui
142  RosUi ui("rgbdslam"); //ui namespace for service calls
143  ui_connections(&ui, &graph_mgr, &listener);//common connections for the user interfaces
144 
145  //If one thread receives a exit signal from the user, signal the other thread to quit too
146  QObject::connect(&app, SIGNAL(aboutToQuit()), &qtRos, SLOT(quitNow()));
147  QObject::connect(&qtRos, SIGNAL(rosQuits()), &app, SLOT(quit()));
148 
149  qtRos.start();// Run main loop.
150  app.exec();
151  //if(ros::ok()) ros::shutdown();//If not yet done through the qt connection
152  //ros::waitForShutdown(); //not sure if necessary.
153 }
