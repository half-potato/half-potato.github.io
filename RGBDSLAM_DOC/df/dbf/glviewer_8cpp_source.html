348 void GLViewer::drawClouds(float xshift) {
349  ScopedTimer s(__FUNCTION__);
350  ParameterServer* ps = ParameterServer::instance();
351  if(follow_mode_){
352  int id = cloud_matrices->size()-1;
353  if(id >= 0)setViewPoint((*cloud_matrices)[id]);
354  }
355  glDisable (GL_BLEND); //Don't blend the clouds, they have no alpha values
356  glLoadIdentity();
357  //Camera transformation
358  glTranslatef(xTra+xshift, yTra, zTra);
359  if(button_pressed_){ //Show axis of left-right movement
360  drawNavigationAxis(0, 0.5, "mouse up/down");
361  }
362  int x_steps = (xRot / 16.0)/rotation_stepping_;
363  glRotatef(x_steps*rotation_stepping_, 1.0, 0.0, 0.0);
364 
365  if(button_pressed_){ //Show axis of left-right movement
366  drawNavigationAxis(1, 0.5, "mouse left/right");
367  }
368  int y_steps = (yRot / 16.0)/rotation_stepping_;
369  glRotatef(y_steps*rotation_stepping_, 0.0, 1.0, 0.0);
370 
371  int z_steps = (zRot / 16.0)/rotation_stepping_;
372  glRotatef(z_steps*rotation_stepping_, 0.0, 0.0, 1.0);
373  if(button_pressed_){
374  drawNavigationAxis(2, 0.5, "ctrl + left/right");
375  }
376 
377  glMultMatrixd(static_cast<GLdouble*>( viewpoint_tf_.data() ));//works as long as qreal and GLdouble are typedefs to double (might depend on hardware)
378  if(show_grid_) {
379  drawGrid(); //Draw a 10x10 grid with 1m x 1m cells
380  }
381  if(show_poses_) drawAxes(0.5);//Show origin as big axis
382 
383  ROS_DEBUG("Drawing %i PointClouds", cloud_list_indices.size());
384  int step = 1;
385  if(button_pressed_ || non_interactive_update_){
386  step = std::max(step,(int)fast_rendering_step_); //if dynamic adapted value is larger: use
387  step = std::max(step,ps->get<int>("fast_rendering_step")); //if fixed positive is larger: Use.
388  }
389  int last_cloud = std::min(cloud_list_indices.size(), cloud_matrices->size());
390  int first_cloud = 0;
391 
392  //For only viewing a single cloud
393  int specific_cloud = ps->get<int>("show_cloud_with_id");
394  if(specific_cloud >= 0){
395  drawOneCloud(specific_cloud);
396  }
397  else //Show all
398  {
399  int i = 0;
400  for(; i < last_cloud - 10; i+=step){
401  ROS_DEBUG("Drawing %d. PointCloud", i);
402  drawOneCloud(i);
403  }
404  for(int j = std::max(last_cloud - 10, 0); j < last_cloud; j++){
405  ROS_DEBUG("Drawing %d. PointCloud.", j);
406  drawOneCloud(j);
407  }
408 
409  }
410 
411  glDisable(GL_DEPTH_TEST);
412  if(show_edges_) drawEdges();
413 
414  for(int i = 0; i<cloud_list_indices.size() && i<cloud_matrices->size(); i++){
415  glPushMatrix();
416  glMultMatrixd(static_cast<GLdouble*>( (*cloud_matrices)[i].data() ));//works as long as qreal and GLdouble are typedefs to double (might depend on hardware)
417  if(show_poses_) drawAxes((i + 1 == cloud_list_indices.size()) ? 0.5:0.075); //Draw last pose Big
418  if(show_ids_) {
419  glColor4f(1-bg_col_[0],1-bg_col_[1],1-bg_col_[2],1.0); //inverse of bg color
420  this->renderText(0.,0.,0.,QString::number(i), QFont("Monospace", 8));
421  }
422  glPopMatrix();
423  }
424  glEnable(GL_DEPTH_TEST);
425 }

776 void GLViewer::pointCloud2GLStrip(pointcloud_type * pc){
777  ScopedTimer s(__FUNCTION__);
778  ROS_DEBUG("Making GL list from point-cloud pointer %p in thread %d", pc, (unsigned int)QThread::currentThreadId());
779  GLuint cloud_list_index = glGenLists(1);
780  if(!cloud_list_index) {
781  ROS_ERROR("No display list could be created");
782  return;
783  }
784  const float mesh_thresh = ParameterServer::instance()->get<double>("squared_meshing_threshold");
785  const float max_depth = ParameterServer::instance()->get<double>("maximum_depth");
786  glNewList(cloud_list_index, GL_COMPILE);
787  //ROS_INFO_COND(!pc->is_dense, "Expected dense cloud for opengl drawing");
788  point_type origin;
789  origin.x = 0;
790  origin.y = 0;
791  origin.z = 0;
792 
793  float depth;
794  bool strip_on = false, flip = false; //if flip is true, first the lower then the upper is inserted
795  const int w=pc->width, h=pc->height;
796  unsigned char b,g,r;
797  const int step = ParameterServer::instance()->get<int>("visualization_skip_step");
798  for( int y = 0; y < h-step; y+=step){ //go through every point and make two triangles
799  for( int x = 0; x < w-step; x+=step){//for it and its neighbours right and/or down
800  using namespace pcl;
801  if(!strip_on){ //Generate vertices for new triangle
802  const point_type* ll = &pc->points[(x)+(y+step)*w]; //one down (lower left corner)
803  if(!validXYZ(*ll, max_depth)) continue; // both new triangles in this step would use this point
804  const point_type* ur = &pc->points[(x+step)+y*w]; //one right (upper right corner)
805  if(!validXYZ(*ur, max_depth)) continue; // both new triangles in this step would use this point
806 
807  const point_type* ul = &pc->points[x+y*w]; //current point (upper right)
808  if(validXYZ(*ul, max_depth)){ //ul, ur, ll all valid
809  depth = squaredEuclideanDistance(*ul,origin);
810  if (squaredEuclideanDistance(*ul,*ll)/depth <= mesh_thresh and
811  squaredEuclideanDistance(*ul,*ll)/depth <= mesh_thresh and
812  squaredEuclideanDistance(*ur,*ll)/depth <= mesh_thresh){
813  glBegin(GL_TRIANGLE_STRIP);
814  strip_on = true;
815  flip = false; //correct order, upper first
816  //Prepare the first two vertices of a triangle strip
817  //drawTriangle(*ul, *ll, *ur);
818  setGLColor(*ul);
819 
820  //glColor3ub(255,0,0);
821  glVertex3f(ul->x, ul->y, ul->z);
822  }
823  }
824  if(!strip_on) { //can't use the point on the upper left, should I still init a triangle?
825  const point_type* lr = &pc->points[(x+step)+(y+step)*w]; //one right-down (lower right)
826  if(!validXYZ(*lr, max_depth)) {
827  //if this is not valid, there is no way to make a new triangle in the next step
828  //and one could have been drawn starting in this step, only if ul had been valid
829  x++;
830  continue;
831  } else { //at least one can be started at the lower left
832  depth = squaredEuclideanDistance(*ur,origin);
833  if (squaredEuclideanDistance(*ur,*ll)/depth <= mesh_thresh and
834  squaredEuclideanDistance(*lr,*ll)/depth <= mesh_thresh and
835  squaredEuclideanDistance(*ur,*lr)/depth <= mesh_thresh){
836  glBegin(GL_TRIANGLE_STRIP);
837  strip_on = true;
838  flip = true; //but the lower has to be inserted first, for correct order
839  }
840  }
841  }
842  if(strip_on) { //Be this the second or the first vertex, insert it
843  setGLColor(*ll);
844 
845  //glColor3ub(0,255,0);
846  glVertex3f(ll->x, ll->y, ll->z);
847  }
848  continue; //not relevant but demonstrate that nothing else is done in this iteration
849  } // end strip was off
850  else
851  {//neighbours to the left and left down are already set
852  const point_type* ul;
853  if(flip){ ul = &pc->points[(x)+(y+step)*w]; } //one down (lower left corner)
854  else { ul = &pc->points[x+y*w]; } //current point (upper right)
855  if(validXYZ(*ul, max_depth)){ //Neighbours to the left are prepared
856  depth = squaredEuclideanDistance(*ul,origin);
857  if (squaredEuclideanDistance(*ul,*(ul-step))/depth > mesh_thresh){
858  glEnd();
859  strip_on = false;
860  continue;
861  }
862  //Complete the triangle with both leftern neighbors
863  //drawTriangle(*ul, *ll, *ur);
864  setGLColor(*ul);
865 
866  //glColor3ub(255,0,0);
867  glVertex3f(ul->x, ul->y, ul->z);
868  } else {
869  glEnd();
870  strip_on = false;
871  continue; //TODO: Could restart with next point instead
872  }
873  //The following point connects one to the left with the other on this horizontal level
874  const point_type* ll;
875  if(flip){ ll = &pc->points[x+y*w]; } //current point (upper right)
876  else { ll = &pc->points[(x)+(y+step)*w]; } //one down (lower left corner)
877  if(validXYZ(*ll, max_depth)){
878  depth = squaredEuclideanDistance(*ll,origin);
879  if (squaredEuclideanDistance(*ul,*ll)/depth > mesh_thresh or
880  squaredEuclideanDistance(*ul,*(ul-step))/depth > mesh_thresh or
881  squaredEuclideanDistance(*ll,*(ll-step))/depth > mesh_thresh){
882  glEnd();
883  strip_on = false;
884  continue;
885  }
886  setGLColor(*ul);
887 
888  glVertex3f(ll->x, ll->y, ll->z);
889  } else {
890  glEnd();
891  strip_on = false;
892  continue;
893  }
894  }//completed triangles if strip is running
895  }
896  if(strip_on) glEnd();
897  strip_on = false;
898  }
899  ROS_DEBUG("Compiled pointcloud into list %i", cloud_list_index);
900  glEndList();
901  cloud_list_indices.push_back(cloud_list_index);
902  //pointcloud_type pc_empty;
903  //pc_empty.points.swap(pc->points);
904  //pc->width = 0;
905  //pc->height = 0;
