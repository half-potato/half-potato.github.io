386 void Graphical_UI::setParam(QString text){
387  std::map<std::string, boost::any>& config = ParameterServer::instance()->getConfigData();
388  std::map<std::string, boost::any>::const_iterator itr;
389  for (itr = config.begin(); itr != config.end(); ++itr) {
390  if(itr->first == qPrintable(text)){
391  bool ok;
392  if (itr->second.type() == typeid(std::string)) {
393  QString prev_val = boost::any_cast<std::string>(itr->second).c_str();
394  QString new_val = QInputDialog::getText(this, text, tr("New Value:"),
395  QLineEdit::Normal, prev_val, &ok);
396  if (ok){
397  ParameterServer::instance()->set(itr->first, std::string(qPrintable(new_val)));
398  }
399  }
400  else if (itr->second.type() == typeid(int)) {
401  int prev_val = boost::any_cast<int>(itr->second);
402  int new_val = QInputDialog::getInt(this, text, tr("New Value:"),
403  prev_val,
404  std::numeric_limits<int>::min(),
405  std::numeric_limits<int>::max(), 1, &ok);
406  if (ok){
407  ParameterServer::instance()->set(itr->first, new_val);
408  }
409  }
410  else if (itr->second.type() == typeid(double)) {
411  double prev_val = boost::any_cast<double>(itr->second);
412  double new_val = QInputDialog::getDouble(this, text, tr("New Value:"),
413  prev_val,
414  std::numeric_limits<double>::min(),
415  std::numeric_limits<double>::max(), 3, &ok);
416  if (ok){
417  ParameterServer::instance()->set(itr->first, new_val);
418  }
419  }
420  else if (itr->second.type() == typeid(bool)) {
421  QStringList truefalse;
422  truefalse.append("False");//item 0
423  truefalse.append("True"); //item 1
424  //Per definition the cast from bool to int results in 0 or 1
425  int prev_val = boost::any_cast<bool>(itr->second);
426  QString new_val = QInputDialog::getItem(this, text, tr("New Value:"),
427  truefalse, prev_val, false, &ok);
428  if (ok){
429  if(new_val == "False")
430  ParameterServer::instance()->set(itr->first, false);
431  else
432  ParameterServer::instance()->set(itr->first, true);
433  }
434  }
435  }
436  }
437 }
