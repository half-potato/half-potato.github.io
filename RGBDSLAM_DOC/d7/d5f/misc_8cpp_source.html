183 void transformAndAppendPointCloud (const pointcloud_type &cloud_in,
184  pointcloud_type &cloud_to_append_to,
185  const tf::Transform transformation, float max_Depth, int )
186 {
187  bool compact = !ParameterServer::instance()->get<bool>("preserve_raster_on_save");
188  Eigen::Matrix4f eigen_transform;
189  pcl_ros::transformAsMatrix(transformation, eigen_transform);
190  unsigned int cloud_to_append_to_original_size = cloud_to_append_to.size();
191  if(cloud_to_append_to.points.size() ==0){
192  cloud_to_append_to.header = cloud_in.header;
193  cloud_to_append_to.width = 0;
194  cloud_to_append_to.height = 0;
195  cloud_to_append_to.is_dense = false;
196  }
197 
198  ROS_DEBUG("max_Depth = %f", max_Depth);
199  ROS_DEBUG("cloud_to_append_to_original_size = %i", cloud_to_append_to_original_size);
200 
201  //Append all points untransformed. This also copies other fields, e.g rgb
202  cloud_to_append_to += cloud_in;
203 
204  Eigen::Matrix3f rot = eigen_transform.block<3, 3> (0, 0);
205  Eigen::Vector3f trans = eigen_transform.block<3, 1> (0, 3);
206  point_type origin = point_type();
207  origin.x = 0;
208  origin.y = 0;
209  origin.z = 0;
210  int j = 0; //Output index
211  for (size_t i = 0; i < cloud_in.points.size (); ++i) //i: Input index
212  {
213  Eigen::Map<Eigen::Vector3f> p_in (const_cast<float*>(&cloud_in.points[i].x), 3, 1);
214  Eigen::Map<Eigen::Vector3f> p_out (&cloud_to_append_to.points[j+cloud_to_append_to_original_size].x, 3, 1);
215  if(compact){ cloud_to_append_to.points[j+cloud_to_append_to_original_size] = cloud_in.points[i]; }
216  //filter out points with a range greater than the given Parameter or do nothing if negativ
217  if(max_Depth >= 0){
218  if(pcl::squaredEuclideanDistance(cloud_in.points[i], origin) > max_Depth*max_Depth){
219  p_out[0]= std::numeric_limits<float>::quiet_NaN();
220  p_out[1]= std::numeric_limits<float>::quiet_NaN();
221  p_out[2]= std::numeric_limits<float>::quiet_NaN();
222  if(!compact) j++;
223  continue;
224  }
225  }
226  if (pcl_isnan (cloud_in.points[i].x) || pcl_isnan (cloud_in.points[i].y) || pcl_isnan (cloud_in.points[i].z)){
227  if(!compact) j++; //Skip, but leave output as is
228  continue;
229  }
230  p_out = rot * p_in + trans;
231  j++;
232  }
233  if(compact){
234  cloud_to_append_to.points.resize(j+cloud_to_append_to_original_size);
235  cloud_to_append_to.width = 1;
236  cloud_to_append_to.height = j+cloud_to_append_to_original_size;
237  }
238 }
