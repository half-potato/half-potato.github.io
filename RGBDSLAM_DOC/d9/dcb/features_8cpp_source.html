43 StatefulFeatureDetector* adjustedGridWrapper(cv::Ptr<DetectorAdjuster> detadj)
44 {
45  ParameterServer* params = ParameterServer::instance();
46  //Try to get between "max" keypoints and 1.5 times of that.
47  //We actually want exactly "max_keypoints" keypoints. Therefore
48  //it's better to overshoot and then cut away the excessive keypoints
49  int min = params->get<int>("max_keypoints"); //Shall not get below max
50  int max = min * 1.5; //
51 
52  int gridRes = params->get<int>("detector_grid_resolution");
53  int gridcells = gridRes*gridRes;
54  int gridmin = round(min/static_cast<float>(gridcells));
55  int gridmax = round(max/static_cast<float>(gridcells));
56  ROS_INFO("Using gridded keypoint detector with %dx%d cells, keeping %d keypoints in total.", gridRes, gridRes, max);
57 
58  StatefulFeatureDetector* detector = adjusterWrapper(detadj, gridmin, gridmax);
59 
60  return new VideoGridAdaptedFeatureDetector(detector, max, gridRes, gridRes);
61 }

64 FeatureDetector* createDetector(const std::string& detectorName){
65  //For anything but SIFTGPU
66  DetectorAdjuster* detAdj = NULL;
67 
68  ROS_INFO_STREAM("Using " << detectorName << " keypoint detector.");
69  if( detectorName == "SIFTGPU" ) {
70  return NULL;// Will not be used
71  }
72  else if(detectorName == "FAST") {
73  detAdj = new DetectorAdjuster("FAST", 20);
74  }
75 #ifdef CV_NONFREE
76  else if(detectorName == "SURF" || detectorName == "SURF128") {
77  detAdj = new DetectorAdjuster("SURF", 200);
78  }
79  else if(detectorName == "SIFT") {
80  detAdj = new DetectorAdjuster("SIFT", 0.04, 0.0001);
81  }
82 #else
83  else if(detectorName == "SURF" || detectorName == "SURF128" || detectorName == "SIFT")
84  {
85  ROS_ERROR("OpenCV non-free functionality (%s) not built in.", detectorName.c_str());
86  ROS_ERROR("To enable non-free functionality build with CV_NONFREE set.");
87  ROS_WARN("Using ORB feature detection instead.");
88  ParameterServer::instance()->set("feature_detector_type", std::string("ORB"));
89  detAdj = new DetectorAdjuster("AORB", 20);
90  }
91 #endif
92  else if(detectorName == "ORB") {
93  detAdj = new DetectorAdjuster("AORB", 20);
94  }
95  else {
96  ROS_ERROR("Unsupported Keypoint Detector. Using ORB as fallback.");
97  return createDetector("ORB");
98  }
99  assert(detAdj != NULL && "No valid detector aduster");
100 
101  ParameterServer* params = ParameterServer::instance();
102  bool gridWrap = (params->get<int>("detector_grid_resolution") > 1);
103  bool dynaWrap = (params->get<int>("adjuster_max_iterations") > 0);
104 
105  if(dynaWrap && gridWrap){
106  return adjustedGridWrapper(detAdj);
107  }
108  else if(dynaWrap){
109  int min = params->get<int>("max_keypoints");
110  int max = min * 1.5; //params->get<int>("max_keypoints");
111  return adjusterWrapper(detAdj, min, max);
112  }
113  else return detAdj;
114 }
