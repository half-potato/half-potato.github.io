109  std::cerr << "To enable non-free functionality build with CV_NONFREE set.";

189 void VideoDynamicAdaptedFeatureDetector::detectImpl(const cv::Mat& _image, std::vector<KeyPoint>& keypoints, const cv::Mat& _mask) const
190 {
191  //In contraast to the original, no oscillation testing is needed as
192  //the loop is broken out of anyway, if too many features were found.
193 
194  //break if the desired number hasn't been reached.
195  int iter_count = escape_iters_;
196  bool checked_for_non_zero_mask = false;
197 
198  do { // detect at least once
199  keypoints.clear();
200 
201  //the adjuster takes care of calling the detector with updated parameters
202  adjuster_->detect(_image, keypoints,_mask);
203  //ROS_INFO("Detected %zu keypoints", keypoints.size());
204  int found_keypoints = static_cast<int>(keypoints.size());
205  if(found_keypoints < min_features_ )
206  {
207  adjuster_->tooFew(min_features_, found_keypoints);
208  //Specific to depth images
209  if(found_keypoints == 0 && !checked_for_non_zero_mask){
210  checked_for_non_zero_mask = true;
211  if(!hasNonZero(_mask)){
212  std::cout << ("Breaking detection iterations, because of missing depth");
213  break; //does not help to iterate if no points have depth
214  }
215  }
216  }
217  else if( int(keypoints.size()) > max_features_ )
218  {
219  adjuster_->tooMany(max_features_, (int)keypoints.size());
220  break;//FIXME: Too many is ok as they are clipped anyway?
221  }
222  else
223  break;
224 
225  iter_count--;
226  } while( iter_count > 0 && adjuster_->good() );
227 
228 }

270 static void aggregateKeypointsPerGridCell(std::vector<std::vector<cv::KeyPoint> >& sub_keypoint_vectors, //will be modified to global coordinates
271  std::vector<cv::KeyPoint>& keypoints_out, //output
272  cv::Size gridSize,
273  cv::Size imageSize,
274  int edgeThreshold)
275 {
276  for(int i = 0; i < gridSize.height; ++i)
277  {
278  int rowstart = std::max((i*imageSize.height)/gridSize.height - edgeThreshold, 0);
279  for( int j = 0; j < gridSize.width; ++j )
280  {
281  int colstart = std::max((j*imageSize.width)/gridSize.width - edgeThreshold, 0);
282 
283  std::vector<cv::KeyPoint>& cell_keypoints = sub_keypoint_vectors[j+i*gridSize.width];
284  std::vector<cv::KeyPoint>::iterator it = cell_keypoints.begin(), end = cell_keypoints.end();
285  for( ; it != end; ++it )
286  {
287  it->pt.x += colstart;
288  it->pt.y += rowstart;
289  }
290  keypoints_out.insert(keypoints_out.end(), cell_keypoints.begin(), cell_keypoints.end());
291  }
292  }

296 void VideoGridAdaptedFeatureDetector::detectImpl( const cv::Mat& image, std::vector<cv::KeyPoint>& keypoints, const cv::Mat& mask ) const
297 {
298  std::vector<std::vector<cv::KeyPoint> > sub_keypoint_vectors(gridCols*gridRows);
299  keypoints.reserve(maxTotalKeypoints);
300  int maxPerCell = maxTotalKeypoints / (gridRows * gridCols);
301 #pragma omp parallel for
302  for( int i = 0; i < gridRows; ++i )
303  {
304  int rowstart = std::max((i*image.rows)/gridRows - edgeThreshold, 0);
305  int rowend = std::min(image.rows, ((i+1)*image.rows)/gridRows + edgeThreshold);
306  cv::Range row_range(rowstart, rowend);
307 #pragma omp parallel for
308  for( int j = 0; j < gridCols; ++j )
309  {
310  int colstart = std::max((j*image.cols)/gridCols - edgeThreshold, 0);
311  int colend = std::min(image.cols, ((j+1)*image.cols)/gridCols + edgeThreshold);
312  cv::Range col_range(colstart, colend);
313  cv::Mat sub_image = image(row_range, col_range);
314  cv::Mat sub_mask;
315  if( !mask.empty()){
316  sub_mask = mask(row_range, col_range);
317  }
318 
319  std::vector<cv::KeyPoint>& sub_keypoints = sub_keypoint_vectors[j+i*gridCols];
320  detectors[j+i*gridCols]->detect( sub_image, sub_keypoints, sub_mask );
321  keepStrongest( maxPerCell, sub_keypoints );
322  }
323  }
324  aggregateKeypointsPerGridCell(sub_keypoint_vectors, keypoints, cv::Size(gridCols, gridRows), image.size(), edgeThreshold);
325 }
