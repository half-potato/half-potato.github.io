65 {
66  for (size_t i = 0; i < msg->transforms.size(); i++)
67  {

71  continue;//HACK for non-tree tf-graph of rgbd_benchmark
72  }
73 
74  try
75  {
76  tf::StampedTransform stTrans;
77  tf::transformStampedMsgToTF(msg->transforms[i], stTrans);

79  transformer->setTransform(stTrans);
80  ROS_DEBUG("Set transform from %s to %s with timestamp %20.10f",
81  msg->transforms[i].header.frame_id.c_str(),
82  msg->transforms[i].child_frame_id.c_str(),
83  stTrans.stamp_.toSec());

217 void OpenNIListener::processBagfile(std::string filename,
218  const std::string& visua_tpc,
219  const std::string& depth_tpc,
220  const std::string& points_tpc,
221  const std::string& cinfo_tpc,
222  const std::string& odom_tpc,
223  const std::string& tf_tpc)
224 {
225  ROS_INFO_NAMED("OpenNIListener", "Loading Bagfile %s", filename.c_str());
226  Q_EMIT iamBusy(4, "Loading Bagfile", 0);
227  { //bag will be destructed after this block (hopefully frees memory for the optimizer)
228  rosbag::Bag input_bag;
229  try{
230  input_bag.open(filename, rosbag::bagmode::Read);
231  } catch(rosbag::BagIOException ex) {
232  ROS_FATAL("Opening Bagfile %s failed: %s Quitting!", filename.c_str(), ex.what());
233  ros::shutdown();
234  return;
235  }
236  ROS_INFO_NAMED("OpenNIListener", "Opened Bagfile %s", filename.c_str());
237 
238  std::vector<std::string> topics;
239  topics = createTopicsVector(visua_tpc, depth_tpc, points_tpc, cinfo_tpc, odom_tpc, tf_tpc);
240  rosbag::View view(input_bag, rosbag::TopicQuery(topics));
241  Q_EMIT iamBusy(4, "Processing Bagfile", view.size());
242  // Simulate sending of the messages in the bagfile
243  std::deque<sensor_msgs::Image::ConstPtr> vis_images;
244  std::deque<sensor_msgs::Image::ConstPtr> dep_images;
245  std::deque<sensor_msgs::CameraInfo::ConstPtr> cam_infos;
246  std::deque<sensor_msgs::PointCloud2::ConstPtr> pointclouds;
247  std::deque<nav_msgs::OdometryConstPtr> odometries;
248  //ros::Time last_tf=ros::Time(0);
249  ros::Time last_tf=ros::TIME_MIN;
250 
251 
252 
253  bool tf_available = false;
254  int counter = 0;
255  BOOST_FOREACH(rosbag::MessageInstance const m, view)
256  {
257  Q_EMIT progress(4, "Processing Bagfile", counter++);
258  do{
259  usleep(10);
260  if(!ros::ok()) return;
261  } while(pause_);
262  ROS_INFO_NAMED("OpenNIListener", "Processing %s of type %s with timestamp %f", m.getTopic().c_str(), m.getDataType().c_str(), m.getTime().toSec());
263 
264  if (m.getTopic() == odom_tpc || ("/" + m.getTopic() == odom_tpc)) {
265  ROS_INFO_NAMED("OpenNIListener", "Processing %s of type %s with timestamp %f", m.getTopic().c_str(), m.getDataType().c_str(), m.getTime().toSec());
266  nav_msgs::OdometryConstPtr odommsg = m.instantiate<nav_msgs::Odometry>();
267  if (odommsg) odometries.push_back(odommsg);
268  }
269  else if (m.getTopic() == visua_tpc || ("/" + m.getTopic() == visua_tpc))
270  {
271  sensor_msgs::Image::ConstPtr rgb_img = m.instantiate<sensor_msgs::Image>();
272  if (rgb_img) vis_images.push_back(rgb_img);
273  ROS_DEBUG("Found Message of %s", visua_tpc.c_str());
274  }
275  else if (m.getTopic() == depth_tpc || ("/" + m.getTopic() == depth_tpc))
276  {
277  sensor_msgs::Image::ConstPtr depth_img = m.instantiate<sensor_msgs::Image>();
278  //if (depth_img) depth_img_sub_->newMessage(depth_img);
279  if (depth_img) dep_images.push_back(depth_img);
280  ROS_DEBUG("Found Message of %s", depth_tpc.c_str());
281  }
282  else if (m.getTopic() == points_tpc || ("/" + m.getTopic() == points_tpc))
283  {
284  sensor_msgs::PointCloud2::ConstPtr pointcloud = m.instantiate<sensor_msgs::PointCloud2>();
285  //if (cam_info) cam_info_sub_->newMessage(cam_info);
286  if (pointcloud) pointclouds.push_back(pointcloud);
287  ROS_DEBUG("Found Message of %s", points_tpc.c_str());
288  }
289  else if (m.getTopic() == cinfo_tpc || ("/" + m.getTopic() == cinfo_tpc))
290  {
291  sensor_msgs::CameraInfo::ConstPtr cam_info = m.instantiate<sensor_msgs::CameraInfo>();
292  //if (cam_info) cam_info_sub_->newMessage(cam_info);
293  if (cam_info) cam_infos.push_back(cam_info);
294  ROS_DEBUG("Found Message of %s", cinfo_tpc.c_str());
295  }
296  else if (m.getTopic() == tf_tpc|| ("/" + m.getTopic() == tf_tpc)){
297  tf::tfMessage::ConstPtr tf_msg = m.instantiate<tf::tfMessage>();
298  if (tf_msg) {
299  tf_available = true;
300  addTFMessageDirectlyToTransformer(tf_msg, tflistener_);
301  last_tf = tf_msg->transforms[0].header.stamp;
302  last_tf -= ros::Duration(0.1);
303  }
304  }
305  if (last_tf == ros::TIME_MIN){//If not a valid time yet, set to something before first message's stamp
306  last_tf = m.getTime();
307  last_tf -= ros::Duration(0.1);
308  }
309  //last_tf = m.getTime();//FIXME: No TF -> no processing
310  while(!odometries.empty() && odometries.front()->header.stamp < last_tf){
311  ROS_INFO_NAMED("OpenNIListener", "Sending Odometry message");
312  odomCallback(odometries.front());
313  odometries.pop_front();
314  }
315  while(!vis_images.empty() && vis_images.front()->header.stamp < last_tf){
316  ROS_INFO_NAMED("OpenNIListener", "Forwarding buffered visual message from time %12f", vis_images.front()->header.stamp.toSec());
317  rgb_img_sub_->newMessage(vis_images.front());
318  vis_images.pop_front();
319  }
320  while(!dep_images.empty() && dep_images.front()->header.stamp < last_tf){
321  ROS_INFO_NAMED("OpenNIListener", "Forwarding buffered depth message from time %12f", dep_images.front()->header.stamp.toSec());
322  depth_img_sub_->newMessage(dep_images.front());
323  dep_images.pop_front();
324  }
325  while(!cam_infos.empty() && cam_infos.front()->header.stamp < last_tf){
326  ROS_INFO_NAMED("OpenNIListener", "Forwarding buffered cam info message from time %12f", cam_infos.front()->header.stamp.toSec());
327  cam_info_sub_->newMessage(cam_infos.front());
328  cam_infos.pop_front();
329  }
330  while(!pointclouds.empty() && pointclouds.front()->header.stamp < last_tf){
331  pc_sub_->newMessage(pointclouds.front());
332  pointclouds.pop_front();
333  }
334 
335  }
336  ROS_WARN_NAMED("eval", "Finished processing of Bagfile");
337  input_bag.close();
338  }

916  rgba_buffers_.resize(idx+1);
