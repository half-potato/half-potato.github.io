171 void GlobalUtil::SetDeviceParam(int argc, char** argv)
172 {
173  if(GlobalParam::_GoodOpenGL!= -1) return;
174 
175  #define CHAR1_TO_INT(x) ((x >= 'A' && x <= 'Z') ? x + 32 : x)
176  #define CHAR2_TO_INT(str, i) (str[i] ? CHAR1_TO_INT(str[i]) + (CHAR1_TO_INT(str[i+1]) << 8) : 0)
177  #define CHAR3_TO_INT(str, i) (str[i] ? CHAR1_TO_INT(str[i]) + (CHAR2_TO_INT(str, i + 1) << 8) : 0)
178  #define STRING_TO_INT(str) (CHAR1_TO_INT(str[0]) + (CHAR3_TO_INT(str, 1) << 8))
179 
180  char* arg, * opt;
181  for(int i = 0; i< argc; i++)
182  {
183  arg = argv[i];
184  if(arg == NULL || arg[0] != '-')continue;
185  opt = arg+1;
186 
188  switch( STRING_TO_INT(opt))
189  {
190  case 'w' + ('i' << 8) + ('n' << 16) + ('p' << 24):
191  if(_GoodOpenGL != 2 && i + 1 < argc)
192  {
193  int x =0, y=0;
194  if(sscanf(argv[++i], "%dx%d", &x, &y) == 2)
195  {
196  GlobalParam::_WindowInitX = x;
197  GlobalParam::_WindowInitY = y;
198  }
199  }
200  break;
201  case 'd' + ('i' << 8) + ('s' << 16) + ('p' << 24):
202  if(_GoodOpenGL != 2 && i + 1 < argc)
203  {
204  GlobalParam::_WindowDisplay = argv[++i];
205  }
206  break;
207  case 'c' + ('u' << 8) + ('d' << 16) + ('a' << 24):
208  if(i + 1 < argc)
209  {
210  int device = 0;
211  scanf(argv[++i], "%d", &device) ;
212  GlobalParam::_DeviceIndex = device;
213  }
214  break;
215  default:
216  break;
217  }
218  }

292 int ClockTimer::ClockMS()
293 {
294 #if defined(_WIN32)
295  #if defined(TIMING_BY_CLOCK)
296  return clock() * 1000 / CLOCKS_PER_SEC;
297  #else
298  static int started = 0;
299  static int tstart;
300  if(started == 0)
301  {
302  tstart = timeGetTime();
303  started = 1;
304  return 0;
305  }else
306  {
307  return timeGetTime() - tstart;
308  }
309  #endif
310 #else
311  static int started = 0;
312  static struct timeval tstart;
313  if(started == 0)
314  {
315  gettimeofday(&tstart, NULL);
316  started = 1;
317  return 0;
318  }else
319  {
320  struct timeval now;
321  gettimeofday(&now, NULL) ;
322  return (now.tv_usec - tstart.tv_usec + (now.tv_sec - tstart.tv_sec) * 1000000)/1000;
323  }
324 #endif
325 }

371 void GlobalUtil::InitGLParam(int NotTargetGL)
372 {
373  //IF the OpenGL context passed the check
374  if(GlobalUtil::_GoodOpenGL == 2) return;
375  //IF the OpenGl context failed the check
376  if(GlobalUtil::_GoodOpenGL == 0) return;
377  //IF se use CUDA or OpenCL
378  if(NotTargetGL && !GlobalUtil::_UseSiftGPUEX)
379  {
380  GlobalUtil::_GoodOpenGL = 1;
381  }else
382  {
383  //first time in this function
384  glewInit();
385 
386  GlobalUtil::_GoodOpenGL = 2;
387 
388  const char * vendor = (const char * )glGetString(GL_VENDOR);
389  if(vendor)
390  {
391  GlobalUtil::_IsNvidia = (strstr(vendor, "NVIDIA") !=NULL ? 1 : 0);
392 
393  // Let nVidia compiler to take care of the unrolling.
394  if (GlobalUtil::_IsNvidia) GlobalUtil::_KeepShaderLoop = 1;
395 
396 #ifndef WIN32
397  else if(!strstr(vendor, "ATI") )
398  {
399  // For non-nVidia non-ATI cards...simply assume it is Mesa
400  // Keep the original shader loop, because some of the unrolled
401  // loopes are too large, and it may take too much time to compile
402  GlobalUtil::_KeepShaderLoop = 1;
403  }
404 #endif
405 
406  if(GlobalUtil::_IsNvidia && glewGetExtension("GL_NVX_gpu_memory_info"))
407  {
408  glGetIntegerv(0x9049/*GL_GPU_MEM_INFO_CURRENT_AVAILABLE_MEM_NVX*/, &_MemCapGPU);
409  _MemCapGPU /= (1024);
410  if(GlobalUtil::_verbose) std::cout << "[GPU VENDOR]:\t" << vendor << ' ' <<_MemCapGPU << "MB\n";
411  }else if(strstr(vendor, "ATI") && glewGetExtension("GL_ATI_meminfo"))
412  {
413  int info[4]; glGetIntegerv(0x87FC/*GL_TEXTURE_FREE_MEMORY_ATI*/, info);
414  _MemCapGPU = info[0] / (1024);
415  if(GlobalUtil::_verbose) std::cout << "[GPU VENDOR]:\t" << vendor << ' ' <<_MemCapGPU << "MB\n";
416  }else
417  {
418  if(GlobalUtil::_verbose) std::cout << "[GPU VENDOR]:\t" << vendor << "\n";
419  }
420 
421  }
422  if(GlobalUtil::_IsNvidia == 0 )GlobalUtil::_UseCUDA = 0;
423 
424  if (glewGetExtension("GL_ARB_fragment_shader") != GL_TRUE ||
425  glewGetExtension("GL_ARB_shader_objects") != GL_TRUE ||
426  glewGetExtension("GL_ARB_shading_language_100") != GL_TRUE)
427  {
428  std::cerr << "Shader not supported by your hardware!\n";
429  GlobalUtil::_GoodOpenGL = 0;
430  }
431 
432  if (glewGetExtension("GL_EXT_framebuffer_object") != GL_TRUE)
433  {
434  std::cerr << "Framebuffer object not supported!\n";
435  GlobalUtil::_GoodOpenGL = 0;
436  }
437 
438  if(glewGetExtension("GL_ARB_texture_rectangle")==GL_TRUE)
439  {
440  GLint value;
441  GlobalUtil::_texTarget = GL_TEXTURE_RECTANGLE_ARB;
442  glGetIntegerv(GL_MAX_RECTANGLE_TEXTURE_SIZE_EXT, &value);
