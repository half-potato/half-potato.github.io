55 tf::StampedTransform GraphManager::computeFixedToBaseTransform(Node* node, bool invert)
56 {
57  g2o::VertexSE3* v = dynamic_cast<g2o::VertexSE3*>(optimizer_->vertex(node->vertex_id_));
58  if(!v){
59  ROS_FATAL("Nullpointer in graph at position %i!", node->vertex_id_);
60  throw std::exception();
61  }
62 
63  tf::Transform computed_motion = eigenTransf2TF(v->estimate());//get pose of point cloud w.r.t. first frame's pc
64  tf::Transform base2points = node->getBase2PointsTransform();//get pose of base w.r.t current pc at capture time
65  printTransform("base2points", base2points);
66  printTransform("computed_motion", computed_motion);
67  printTransform("init_base_pose_", init_base_pose_);
68 
69  tf::Transform world2base = init_base_pose_*base2points*computed_motion*base2points.inverse();
70  tf::Transform gt_world2base = node->getGroundTruthTransform();//get mocap pose of base in map
71  tf::Transform err = gt_world2base.inverseTimes(world2base);
72 
73  //makes sure things have a corresponding timestamp
74  //also avoids problems with tflistener cache size if mapping took long. Must be synchronized with tf broadcasting
75  ros::Time now = ros::Time::now();
76 
77  printTransform("World->Base", world2base);
78  std::string fixed_frame = ParameterServer::instance()->get<std::string>("fixed_frame_name");
79  std::string base_frame = ParameterServer::instance()->get<std::string>("base_frame_name");
80  if(base_frame.empty())
81  { //if there is no base frame defined, use frame of sensor data
82  base_frame = graph_.begin()->second->header_.frame_id;
83  }
84  if(invert){
85  return tf::StampedTransform(world2base.inverse(), now, base_frame, fixed_frame);
86  } else {
87  return tf::StampedTransform(world2base, now, fixed_frame, base_frame);
88  }
