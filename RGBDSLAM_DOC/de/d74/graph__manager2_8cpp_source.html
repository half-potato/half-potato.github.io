111 void GraphManager::setEmpiricalCovariancesForEdgeSet(EdgeSet& edges){
112  ScopedTimer s(__FUNCTION__);
113 
114  typedef Eigen::Matrix<double,6,Eigen::Dynamic> Matrix6Xd;
115  typedef g2o::BaseEdge<6, Eigen::Isometry3d> BaseEdgeSE3;
116  typedef g2o::EdgeSE3::InformationType InfoMat;
117 
118  Matrix6Xd measurements, errors;
119  edgesToMeasurementMatrix<BaseEdgeSE3>(edges, measurements);
120  edgesToErrorMatrix<BaseEdgeSE3>(edges, errors);
121 
122  //Precomputation of weights and weighted errors
123  Vector6d meanMsr = measurements.rowwise().mean();
124  Vector6d stdDev = (measurements.colwise() - meanMsr).cwiseAbs().rowwise().mean();
125 
126  EdgeSet::iterator it = edges.begin();
127  for(int i = 0; it != edges.end(); ++it, ++i)
128  {
129  BaseEdgeSE3* edge = dynamic_cast<BaseEdgeSE3*>( *it );
130  if(edge){
131  InfoMat infoMat = computeEmpiricalInformationMatrix(measurements, errors, measurements.col(i), stdDev);
132  edge->setInformation(infoMat);
133  }
134  }
135 
136 }
