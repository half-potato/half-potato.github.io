113 void SiftGPUWrapper::detect(const cv::Mat& image, cv::vector<cv::KeyPoint>& keypoints, std::vector<float>& descriptors, const Mat& mask) const {
114  ScopedTimer s(__FUNCTION__);
115  if (error) {
116  keypoints.clear();
117  ROS_FATAL("SiftGPU cannot be used. Detection of keypoints failed");
118  }
119 
120  //get image
121  if(image.rows != imageHeight || image.cols != imageWidth){
122  imageHeight = image.rows;
123  imageWidth = image.cols;
124  free(data);
125  data = (unsigned char*) malloc(imageWidth * imageHeight);
126  }
127  cvMatToSiftGPU(image, data);
128 
129  int num_features = 0;
130  SiftGPU::SiftKeypoint* keys = 0;
131 
133  if(keypoints.size() != 0){
134  keys = new SiftGPU::SiftKeypoint[keypoints.size()];
135  for(int i = 0; i < keypoints.size(); ++i){
136  keys[i].x = keypoints[i].pt.x;
137  keys[i].y = keypoints[i].pt.y;
138  keys[i].o = keypoints[i].angle / 180.0 * 3.1415927;//opencv uses degree
139  keys[i].s = keypoints[i].size / 12.0;//Changchang Wu: "You can use 6*scale as the radius" and OpenCV uses diameter
140  }
141  siftgpu->SetKeypointList(keypoints.size(), &keys[0]);
142  }
143 
144  gpu_mutex.lock();
145  ROS_DEBUG("SIFTGPU: cols: %d, rows: %d", image.cols, image.rows);
146  if (siftgpu->RunSIFT(image.cols, image.rows, data, GL_LUMINANCE, GL_UNSIGNED_BYTE)) {
147  num_features = siftgpu->GetFeatureNum();
148  ROS_DEBUG("Number of features found: %i", num_features);
149  descriptors.resize(128 * num_features);
150  //descriptors = new float[128 * num_features];
151  if(keypoints.size() == 0){
152  keys = new SiftGPU::SiftKeypoint[num_features];
153  siftgpu->GetFeatureVector(&keys[0], &descriptors[0]);
154  }
155  siftgpu->GetFeatureVector(NULL, &descriptors[0]);
156  } else {
157  ROS_WARN("SIFTGPU->RunSIFT() failed!");
158  }
159 
160  //copy to opencv structure
161  keypoints.clear();
162  for (int i = 0; i < num_features; ++i) {
163  KeyPoint key(keys[i].x, keys[i].y, 12.0 * keys[i].s, keys[i].o * 180.0 / 3.1415927); // 6 x scale is the conversion to pixels of the radius, according to changchang wu (the author of siftgpu). OpenCV uses Diameter and degree
164  keypoints.push_back(key);
165  }
166  gpu_mutex.unlock();
167 }

169 int SiftGPUWrapper::match(
170  const std::vector<float>& descriptors1,
171  int num1,
172  const std::vector<float>& descriptors2,
173  int num2,
174  std::vector<cv::DMatch>* matches) {
175  if (!isMatcherInitialized)
176  initializeMatcher();
177 
178  float sumDistances = 0;
179 
180  gpu_mutex.lock();
181  matcher->SetDescriptors(0, num1, &descriptors1[0]);
182  matcher->SetDescriptors(1, num2, &descriptors2[0]);
183 
184  int (*match_buf)[2] = new int[num1][2];
185  int number = matcher->GetSiftMatch(num1, match_buf, 0.9, 0.9);
186 
187  if (matches->size() != 0) {
188  ROS_WARN("Clearing matches vector!");
189  matches->clear();
190  }
191 
192  cv::DMatch match;
193  int counter = 0;
194 
195  for (int i = 0; i < number; i++) {
196  match.queryIdx = match_buf[i][0];
197  match.trainIdx = match_buf[i][1];
198 
199  //only use matches with indices != 0 (opengl context problem may sometimes happen)
200  if (match.queryIdx == 0 || match.trainIdx == 0) {
201  counter++;
202  }
203 
204  if (counter > 0.5 * number) {
205  matches->clear();
206  sumDistances = 0;
207  ROS_ERROR("Matches bad due to context error");
208  break;
209  }
210 
211  float sum = 0;
212  for (int j = 0; j < 128; j++) {
213  float a = descriptors1[match.queryIdx * 128 + j] - descriptors2[match.trainIdx * 128 + j];
214  sum += a * a;
215  }
216 
217  match.distance = sqrt(sum);
218  sumDistances += match.distance;
219  matches->push_back(match);
220  ROS_DEBUG("Matched Features %d and %d with distance of %f. Sum: %f", match.queryIdx, match.trainIdx, match.distance, sumDistances);
221  }
222 
223  delete[] match_buf;
224 
225  gpu_mutex.unlock();
226  return sumDistances;
227 }
