132 void ColorOctomapServer::occupancyFilter(pointcloud_type::ConstPtr input,
133  pointcloud_type::Ptr output,
134  double occupancy_threshold){
135  if(output->points.capacity() < input->size()){ //cannot happen for input == output
136  output->reserve(input->size());
137  }
138 
139  Eigen::Quaternionf q = input->sensor_orientation_;
140  Eigen::Vector4f t = input->sensor_origin_;
141 
142  size_t size = input->size();
143  size_t outidx = 0;
144  for (size_t inidx = 0; inidx < size; ++inidx){
145  const point_type& in_point = (*input)[inidx];
146  Eigen::Vector3f in_vec = q * in_point.getVector3fMap() + t.head<3>();
147  if (std::isnan(in_vec.z()))
148  continue;
149 
150  const int radius = 1;
151  int x_a = m_octoMap.coordToKey(in_vec.x()) - radius;
152  int x_b = m_octoMap.coordToKey(in_vec.x()) + radius;
153  int y_a = m_octoMap.coordToKey(in_vec.y()) - radius;
154  int y_b = m_octoMap.coordToKey(in_vec.y()) + radius;
155  int z_a = m_octoMap.coordToKey(in_vec.z()) - radius;
156  int z_b = m_octoMap.coordToKey(in_vec.z()) + radius;
157  double sum_of_occupancy = 0, sum_of_weights = 0;
158  for(;x_a <= x_b; ++x_a){
159  for(;y_a <= y_b; ++y_a){
160  for(;z_a <= z_b; ++z_a){
161  octomap::OcTreeNode* node = m_octoMap.search(octomap::OcTreeKey(x_a, y_a, z_a));
162  if(node != NULL){
163  double dx = m_octoMap.keyToCoord(x_a) - in_vec.x();
164  double dy = m_octoMap.keyToCoord(y_a) - in_vec.y();
165  double dz = m_octoMap.keyToCoord(z_a) - in_vec.z();
166  double weight = dx*dx+dy*dy+dz*dz;
167  double weighted_occ = node->getOccupancy() / weight;
168  sum_of_weights += weight;
169  sum_of_occupancy += weighted_occ;
170  }
171  }
172  }
173  }
174 
175 
176  if(sum_of_occupancy < occupancy_threshold * sum_of_weights) //Filters points in non-existent nodes (outside of map?)
177  //if(node != NULL && node->getOccupancy() >= occupancy_threshold)
178  { //Valid point
179  point_type& out_point = (*output)[outidx];
180  out_point = in_point;
181  ++outidx;
182  }
183  }
184  output->resize(outidx);//downsize
185 }
