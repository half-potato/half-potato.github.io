166 int GetMilliSecond()
167 {
168 #if defined(_WIN32)
169  #if defined(TIMING_BY_CLOCK)
170  return clock() * 1000 / CLOCKS_PER_SEC;
171  #else
172  static int started = 0;
173  static int tstart;
174  //the resolution of teimGetTime will be changed to 1ms inside SiftGPU
175  if(started == 0)
176  {
177  tstart = timeGetTime();
178  started = 1;
179  return 0;
180  }else
181  {
182  return timeGetTime() - tstart;
183  }
184  #endif
185 #else
186  static int started = 0;
187  static struct timeval tstart;
188  if(started == 0)
189  {

191  started = 1;
192  return 0;
193  }else
194  {

198  }

200 }
