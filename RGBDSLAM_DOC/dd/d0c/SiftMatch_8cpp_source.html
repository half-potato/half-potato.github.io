249 void SiftMatchGL::LoadSiftMatchShadersGLSL()
250 {
251  ProgramGLSL * program;
252  ostringstream out;
253  if(GlobalUtil::_IsNvidia)
254  out << "#pragma optionNV(ifcvt none)\n"
255  "#pragma optionNV(unroll all)\n";
256 
257  out << "#define SIFT_PER_STRIPE " << _sift_per_stripe << ".0\n"
258  "#define PIXEL_PER_SIFT " << _pixel_per_sift << "\n"
259  "uniform sampler2DRect tex1, tex2; uniform vec2 size;\n"
260  "void main() \n"
261  "{\n"
262  << " vec4 val = vec4(0.0, 0.0, 0.0, 0.0), data1, buf;\n"
263  " vec2 index = gl_FragCoord.yx; \n"
264  " vec2 stripe_size = size.xy * SIFT_PER_STRIPE;\n"
265  " vec2 temp_div1 = index / stripe_size;\n"
266  " vec2 stripe_index = floor(temp_div1);\n"
267  " index = floor(stripe_size * (temp_div1 - stripe_index));\n"
268  " vec2 temp_div2 = index * vec2(1.0 / float(SIFT_PER_STRIPE));\n"
269  " vec2 temp_floor2 = floor(temp_div2);\n"
270  " vec2 index_v = temp_floor2 + vec2(0.5);\n "
271  " vec2 index_h = vec2(SIFT_PER_STRIPE)* (temp_div2 - temp_floor2);\n"
272  " vec2 tx = (index_h + stripe_index * vec2(SIFT_PER_STRIPE))* vec2(PIXEL_PER_SIFT) + 0.5;\n"
273  " vec2 tpos1, tpos2; \n"
274  " vec4 tpos = vec4(tx, index_v);\n"
276  " for(int i = 0; i < PIXEL_PER_SIFT; ++i){\n"
277  " buf = texture2DRect(tex2, tpos.yw);\n"
278  " data1 = texture2DRect(tex1, tpos.xz);\n"
279  " val += (data1 * buf);\n"
280  " tpos.xy = tpos.xy + vec2(1.0, 1.0);\n"
281  " }\n"
282  " const float factor = 0.248050689697265625; \n"
283  " gl_FragColor =vec4(dot(val, vec4(factor)), index, 0);\n"
284  "}"
285  << '\0';
286 
287  s_multiply = program= new ProgramGLSL(out.str().c_str());

292 
293  out.seekp(ios::beg);
294  if(GlobalUtil::_IsNvidia)
295  out << "#pragma optionNV(ifcvt none)\n"
296  "#pragma optionNV(unroll all)\n";
297 
298  out << "#define SIFT_PER_STRIPE " << _sift_per_stripe << ".0\n"
299  "#define PIXEL_PER_SIFT " << _pixel_per_sift << "\n"
300  "uniform sampler2DRect tex1, tex2;\n"
301  "uniform sampler2DRect texL1;\n"
302  "uniform sampler2DRect texL2; \n"
303  "uniform mat3 H; \n"
304  "uniform mat3 F; \n"
305  "uniform vec4 size; \n"
306  "void main() \n"
307  "{\n"
308  << " vec4 val = vec4(0.0, 0.0, 0.0, 0.0), data1, buf;\n"
309  " vec2 index = gl_FragCoord.yx; \n"
310  " vec2 stripe_size = size.xy * SIFT_PER_STRIPE;\n"
311  " vec2 temp_div1 = index / stripe_size;\n"
312  " vec2 stripe_index = floor(temp_div1);\n"
313  " index = floor(stripe_size * (temp_div1 - stripe_index));\n"
314  " vec2 temp_div2 = index * vec2(1.0/ float(SIFT_PER_STRIPE));\n"
315  " vec2 temp_floor2 = floor(temp_div2);\n"
316  " vec2 index_v = temp_floor2 + vec2(0.5);\n "
317  " vec2 index_h = vec2(SIFT_PER_STRIPE)* (temp_div2 - temp_floor2);\n"
318 
319  //read feature location data
320  " vec4 tlpos = vec4((index_h + stripe_index * vec2(SIFT_PER_STRIPE)) + 0.5, index_v);\n"
321  " vec3 loc1 = vec3(texture2DRect(texL1, tlpos.xz).xw, 1.0);\n"
322  " vec3 loc2 = vec3(texture2DRect(texL2, tlpos.yw).xw, 1.0);\n"
323 
324  //check the guiding homography
325  " vec3 hxloc1 = H* loc1;\n"
326  " vec2 diff = abs(loc2.xy- (hxloc1.xy/hxloc1.z));\n"
327  " float disth = max(diff.x, diff.y);\n"
328  " if(disth > size.z ) {gl_FragColor = vec4(0.0, index, 0.0); return;}\n"
329 
330  //check the guiding fundamental
331  " vec3 fx1 = (F * loc1), ftx2 = (loc2 * F);\n"
332  " float x2tfx1 = dot(loc2, fx1);\n"
333  " vec4 temp = vec4(fx1.xy, ftx2.xy); \n"
334  " float sampson_error = (x2tfx1 * x2tfx1) / dot(temp, temp);\n"
335  " if(sampson_error > size.w) {gl_FragColor = vec4(0.0, index, 0.0); return;}\n"
336 
337  //compare feature descriptor
338  " vec2 tx = (index_h + stripe_index * SIFT_PER_STRIPE)* vec2(PIXEL_PER_SIFT) + 0.5;\n"
339  " vec2 tpos1, tpos2; \n"
340  " vec4 tpos = vec4(tx, index_v);\n"
341  " for(int i = 0; i < PIXEL_PER_SIFT; ++i){\n"
342  " buf = texture2DRect(tex2, tpos.yw);\n"
343  " data1 = texture2DRect(tex1, tpos.xz);\n"
344  " val += data1 * buf;\n"
345  " tpos.xy = tpos.xy + vec2(1.0, 1.0);\n"
346  " }\n"
347  " const float factor = 0.248050689697265625; \n"
