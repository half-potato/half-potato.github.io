56 static void
57 HarrisResponses(const Mat& img, vector<KeyPoint>& pts, int blockSize, float harris_k)
58 {
59  CV_Assert( img.type() == CV_8UC1 && blockSize*blockSize <= 2048 );
60 
61  size_t ptidx, ptsize = pts.size();
62 
63  const uchar* ptr00 = img.ptr<uchar>();
64  int step = (int)(img.step/img.elemSize1());
65  int r = blockSize/2;
66 
67  float scale = (1 << 2) * blockSize * 255.0f;
68  scale = 1.0f / scale;
69  float scale_sq_sq = scale * scale * scale * scale;
70 
71  AutoBuffer<int> ofsbuf(blockSize*blockSize);
72  int* ofs = ofsbuf;
73  for( int i = 0; i < blockSize; i++ )
74  for( int j = 0; j < blockSize; j++ )
75  ofs[i*blockSize + j] = (int)(i*step + j);
76 
77  for( ptidx = 0; ptidx < ptsize; ptidx++ )
78  {
79  int x0 = cvRound(pts[ptidx].pt.x - r);
80  int y0 = cvRound(pts[ptidx].pt.y - r);
81 
82  const uchar* ptr0 = ptr00 + y0*step + x0;
83  int a = 0, b = 0, c = 0;
84 
85  for( int k = 0; k < blockSize*blockSize; k++ )
86  {
87  const uchar* ptr = ptr0 + ofs[k];
88  int Ix = (ptr[1] - ptr[-1])*2 + (ptr[-step+1] - ptr[-step-1]) + (ptr[step+1] - ptr[step-1]);
89  int Iy = (ptr[step] - ptr[-step])*2 + (ptr[step-1] - ptr[-step-1]) + (ptr[step+1] - ptr[-step+1]);
90  a += Ix*Ix;
91  b += Iy*Iy;
92  c += Ix*Iy;
93  }
94  pts[ptidx].response = ((float)a * b - (float)c * c -
95  harris_k * ((float)a + b) * ((float)a + b))*scale_sq_sq;
96  }
97 }

101 static float IC_Angle(const Mat& image, const int half_k, Point2f pt,
102  const vector<int> & u_max)
103 {
104  int m_01 = 0, m_10 = 0;
105 
106  const uchar* center = &image.at<uchar> (cvRound(pt.y), cvRound(pt.x));
107 
108  // Treat the center line differently, v=0
109  for (int u = -half_k; u <= half_k; ++u)
110  m_10 += u * center[u];
111 
112  // Go line by line in the circular patch
113  int step = (int)image.step1();
114  for (int v = 1; v <= half_k; ++v)
115  {
116  // Proceed over the two lines
117  int v_sum = 0;
118  int d = u_max[v];
119  for (int u = -d; u <= d; ++u)
120  {
121  int val_plus = center[u + v*step], val_minus = center[u - v*step];
122  v_sum += (val_plus - val_minus);
123  m_10 += u * (val_plus + val_minus);
124  }
125  m_01 += v * v_sum;
126  }
127 
128  return fastAtan2((float)m_01, (float)m_10);
129 }

133 static void computeOrbDescriptor(const KeyPoint& kpt,
134  const Mat& img, const Point* pattern,
135  uchar* desc, int dsize, int WTA_K)
136 {
137  float angle = kpt.angle;
138  //angle = cvFloor(angle/12)*12.f;
139  angle *= (float)(CV_PI/180.f);
140  float a = (float)cos(angle), b = (float)sin(angle);
141 
142  const uchar* center = &img.at<uchar>(cvRound(kpt.pt.y), cvRound(kpt.pt.x));
143  int step = (int)img.step;
144 
145 #if 1
146  #define GET_VALUE(idx) \
147  center[cvRound(pattern[idx].x*b + pattern[idx].y*a)*step + \
148  cvRound(pattern[idx].x*a - pattern[idx].y*b)]
149 #else
150  float x, y;
151  int ix, iy;
152  #define GET_VALUE(idx) \
153  (x = pattern[idx].x*a - pattern[idx].y*b, \
154  y = pattern[idx].x*b + pattern[idx].y*a, \
155  ix = cvFloor(x), iy = cvFloor(y), \
156  x -= ix, y -= iy, \
157  cvRound(center[iy*step + ix]*(1-x)*(1-y) + center[(iy+1)*step + ix]*(1-x)*y + \
158  center[iy*step + ix+1]*x*(1-y) + center[(iy+1)*step + ix+1]*x*y))
159 #endif
160 
161  if( WTA_K == 2 )
162  {
163  for (int i = 0; i < dsize; ++i, pattern += 16)
164  {
165  int t0, t1, val;
166  t0 = GET_VALUE(0); t1 = GET_VALUE(1);
167  val = t0 < t1;
168  t0 = GET_VALUE(2); t1 = GET_VALUE(3);
169  val |= (t0 < t1) << 1;
170  t0 = GET_VALUE(4); t1 = GET_VALUE(5);
171  val |= (t0 < t1) << 2;
172  t0 = GET_VALUE(6); t1 = GET_VALUE(7);
173  val |= (t0 < t1) << 3;
174  t0 = GET_VALUE(8); t1 = GET_VALUE(9);
175  val |= (t0 < t1) << 4;
176  t0 = GET_VALUE(10); t1 = GET_VALUE(11);
177  val |= (t0 < t1) << 5;
178  t0 = GET_VALUE(12); t1 = GET_VALUE(13);
179  val |= (t0 < t1) << 6;
180  t0 = GET_VALUE(14); t1 = GET_VALUE(15);
181  val |= (t0 < t1) << 7;
182 
183  desc[i] = (uchar)val;
184  }
185  }
186  else if( WTA_K == 3 )
187  {
188  for (int i = 0; i < dsize; ++i, pattern += 12)
189  {
190  int t0, t1, t2, val;
191  t0 = GET_VALUE(0); t1 = GET_VALUE(1); t2 = GET_VALUE(2);
192  val = t2 > t1 ? (t2 > t0 ? 2 : 0) : (t1 > t0);
193 
194  t0 = GET_VALUE(3); t1 = GET_VALUE(4); t2 = GET_VALUE(5);
195  val |= (t2 > t1 ? (t2 > t0 ? 2 : 0) : (t1 > t0)) << 2;
196 
197  t0 = GET_VALUE(6); t1 = GET_VALUE(7); t2 = GET_VALUE(8);
198  val |= (t2 > t1 ? (t2 > t0 ? 2 : 0) : (t1 > t0)) << 4;
199 
200  t0 = GET_VALUE(9); t1 = GET_VALUE(10); t2 = GET_VALUE(11);
201  val |= (t2 > t1 ? (t2 > t0 ? 2 : 0) : (t1 > t0)) << 6;
202 
203  desc[i] = (uchar)val;
204  }
205  }
206  else if( WTA_K == 4 )
207  {
208  for (int i = 0; i < dsize; ++i, pattern += 16)
209  {
210  int t0, t1, t2, t3, u, v, k, val;
211  t0 = GET_VALUE(0); t1 = GET_VALUE(1);
212  t2 = GET_VALUE(2); t3 = GET_VALUE(3);
213  u = 0, v = 2;
214  if( t1 > t0 ) t0 = t1, u = 1;
215  if( t3 > t2 ) t2 = t3, v = 3;
216  k = t0 > t2 ? u : v;
217  val = k;
218 
219  t0 = GET_VALUE(4); t1 = GET_VALUE(5);
220  t2 = GET_VALUE(6); t3 = GET_VALUE(7);
221  u = 0, v = 2;
222  if( t1 > t0 ) t0 = t1, u = 1;
223  if( t3 > t2 ) t2 = t3, v = 3;
224  k = t0 > t2 ? u : v;
225  val |= k << 2;
226 
227  t0 = GET_VALUE(8); t1 = GET_VALUE(9);
228  t2 = GET_VALUE(10); t3 = GET_VALUE(11);
229  u = 0, v = 2;
230  if( t1 > t0 ) t0 = t1, u = 1;
231  if( t3 > t2 ) t2 = t3, v = 3;
232  k = t0 > t2 ? u : v;
233  val |= k << 4;
234 
235  t0 = GET_VALUE(12); t1 = GET_VALUE(13);
236  t2 = GET_VALUE(14); t3 = GET_VALUE(15);
237  u = 0, v = 2;
238  if( t1 > t0 ) t0 = t1, u = 1;
239  if( t3 > t2 ) t2 = t3, v = 3;
240  k = t0 > t2 ? u : v;
241  val |= k << 6;
242 
243  desc[i] = (uchar)val;
244  }
245  }
246  else
247  CV_Error( CV_StsBadSize, "Wrong WTA_K. It can be only 2, 3 or 4." );
248 
249  #undef GET_VALUE
250 }

253 static void initializeOrbPattern( const Point* pattern0, vector<Point>& pattern, int ntuples, int tupleSize, int poolSize )
254 {
255  RNG rng(0x12345678);
256  int i, k, k1;
257  pattern.resize(ntuples*tupleSize);
258 
259  for( i = 0; i < ntuples; i++ )
260  {
261  for( k = 0; k < tupleSize; k++ )
262  {
263  for(;;)
264  {
265  int idx = rng.uniform(0, poolSize);
266  Point pt = pattern0[idx];
267  for( k1 = 0; k1 < k; k1++ )
268  if( pattern[tupleSize*i + k1] == pt )
269  break;
270  if( k1 == k )
271  {
272  pattern[tupleSize*i + k] = pt;
273  break;
274  }
275  }
276  }
277  }

613 static void computeKeyPoints(const vector<Mat>& imagePyramid,
614  const vector<Mat>& maskPyramid,
615  vector<vector<KeyPoint> >& allKeypoints,
616  int nfeatures, int firstLevel, double scaleFactor,
617  int edgeThreshold, int patchSize, int scoreType, int fastThreshold )
618 {
619  int nlevels = (int)imagePyramid.size();
620  vector<int> nfeaturesPerLevel(nlevels);
621 
622  // fill the extractors and descriptors for the corresponding scales
623  float factor = (float)(1.0 / scaleFactor);
624  float ndesiredFeaturesPerScale = nfeatures*(1 - factor)/(1 - (float)pow((double)factor, (double)nlevels));
625 
626  int sumFeatures = 0;
627  for( int level = 0; level < nlevels-1; level++ )
628  {
629  nfeaturesPerLevel[level] = cvRound(ndesiredFeaturesPerScale);
630  sumFeatures += nfeaturesPerLevel[level];
631  ndesiredFeaturesPerScale *= factor;
632  }
633  nfeaturesPerLevel[nlevels-1] = std::max(nfeatures - sumFeatures, 0);
634 
635  // Make sure we forget about what is too close to the boundary
636  //edge_threshold_ = std::max(edge_threshold_, patch_size_/2 + kKernelWidth / 2 + 2);
637 
638  // pre-compute the end of a row in a circular patch
639  int halfPatchSize = patchSize / 2;
640  vector<int> umax(halfPatchSize + 1);
641 
642  int v, v0, vmax = cvFloor(halfPatchSize * sqrt(2.f) / 2 + 1);
643  int vmin = cvCeil(halfPatchSize * sqrt(2.f) / 2);
644  for (v = 0; v <= vmax; ++v)
645  umax[v] = cvRound(sqrt((double)halfPatchSize * halfPatchSize - v * v));
646 
647  // Make sure we are symmetric
648  for (v = halfPatchSize, v0 = 0; v >= vmin; --v)
649  {
650  while (umax[v0] == umax[v0 + 1])
651  ++v0;
652  umax[v] = v0;
653  ++v0;
654  }
655 
656  allKeypoints.resize(nlevels);
657 
658  for (int level = 0; level < nlevels; ++level)
659  {
660  int nfeatures = nfeaturesPerLevel[level];
661  allKeypoints[level].reserve(nfeatures*2);
662 
663  vector<KeyPoint> & keypoints = allKeypoints[level];
664 
665  // Detect FAST features, 20 is not a good threshold for SLAM
666  FastFeatureDetector fd(fastThreshold, true);
667  fd.detect(imagePyramid[level], keypoints, maskPyramid[level]);
668 
669  // Remove keypoints very close to the border
670  KeyPointsFilter::runByImageBorder(keypoints, imagePyramid[level].size(), edgeThreshold);
671 
672  if( scoreType == AORB::HARRIS_SCORE )
673  {
674  // Keep more points than necessary as FAST does not give amazing corners
675  KeyPointsFilter::retainBest(keypoints, 2 * nfeatures);
676 
677  // Compute the Harris cornerness (better scoring than FAST)
678  HarrisResponses(imagePyramid[level], keypoints, 7, HARRIS_K);
679  }
680 
681  //cull to the final desired level, using the new Harris scores or the original FAST scores.
682  KeyPointsFilter::retainBest(keypoints, nfeatures);
683 
684  float sf = getScale(level, firstLevel, scaleFactor);
685 
686  // Set the level of the coordinates
687  for (vector<KeyPoint>::iterator keypoint = keypoints.begin(),
688  keypointEnd = keypoints.end(); keypoint != keypointEnd; ++keypoint)
689  {
690  keypoint->octave = level;
691  keypoint->size = patchSize*sf;
692  }
693 
694  computeOrientation(imagePyramid[level], keypoints, halfPatchSize, umax);
695  }
696 }

728  OutputArray _descriptors, bool useProvidedKeypoints) const
729 {
730  bool do_keypoints = !useProvidedKeypoints;
731  bool do_descriptors = _descriptors.needed();
732 
733  if( (!do_keypoints && !do_descriptors) || _image.empty() )
734  return;
735 
736  //ROI handling
737  const int HARRIS_BLOCK_SIZE = 9;
738  int halfPatchSize = patchSize / 2;
739  int border = std::max(edgeThreshold, std::max(halfPatchSize, HARRIS_BLOCK_SIZE/2))+1;
740 
741  Mat image = _image.getMat(), mask = _mask.getMat();
742  if( image.type() != CV_8UC1 )
743  cvtColor(_image, image, CV_BGR2GRAY);
744 
745  int nlevels = this->nlevels;
746 
747  if( !do_keypoints )
748  {
749  // if we have pre-computed keypoints, they may use more levels than it is set in parameters
750  // !!!TODO!!! implement more correct method, independent from the used keypoint detector.
751  // Namely, the detector should provide correct size of each keypoint. Based on the keypoint size
752  // and the algorithm used (i.e. BRIEF, running on 31x31 patches) we should compute the approximate
753  // scale-factor that we need to apply. Then we should cluster all the computed scale-factors and
754  // for each cluster compute the corresponding image.
755  //
756  // In short, ultimately the descriptor should
757  // ignore octave parameter and deal only with the keypoint size.
758  nlevels = 0;
759  for( size_t i = 0; i < _keypoints.size(); i++ )

761  nlevels++;
762  }

765  vector<Mat> imagePyramid(nlevels), maskPyramid(nlevels);
766  for (int level = 0; level < nlevels; ++level)
767  {
768  float scale = 1/getScale(level, firstLevel, scaleFactor);
769  Size sz(cvRound(image.cols*scale), cvRound(image.rows*scale));
770  Size wholeSize(sz.width + border*2, sz.height + border*2);
771  Mat temp(wholeSize, image.type()), masktemp;
772  imagePyramid[level] = temp(Rect(border, border, sz.width, sz.height));
773 
774  if( !mask.empty() )
775  {

777  maskPyramid[level] = masktemp(Rect(border, border, sz.width, sz.height));
778  }

781  if( level != firstLevel )
782  {
783  if( level < firstLevel )
784  {
785  resize(image, imagePyramid[level], sz, scale, scale, INTER_LINEAR);
786  if (!mask.empty())
787  resize(mask, maskPyramid[level], sz, scale, scale, INTER_LINEAR);
788  copyMakeBorder(imagePyramid[level], temp, border, border, border, border,
789  BORDER_REFLECT_101+BORDER_ISOLATED);
790  }
791  else
792  {
793  resize(imagePyramid[level-1], imagePyramid[level], sz,
794  1./scaleFactor, 1./scaleFactor, INTER_LINEAR);
795  if (!mask.empty())
796  resize(maskPyramid[level-1], maskPyramid[level], sz,
797  1./scaleFactor, 1./scaleFactor, INTER_LINEAR);
798  copyMakeBorder(imagePyramid[level], temp, border, border, border, border,
799  BORDER_REFLECT_101+BORDER_ISOLATED);
800  }
801  }
802  else
803  {
804  copyMakeBorder(image, temp, border, border, border, border,
805  BORDER_REFLECT_101);
806  image.copyTo(imagePyramid[level]);
807  if( !mask.empty() )
808  mask.copyTo(maskPyramid[level]);
809  }
810 
811  if( !mask.empty() )
812  copyMakeBorder(maskPyramid[level], masktemp, border, border, border, border,
813  BORDER_CONSTANT+BORDER_ISOLATED);
814  }
815 
816  // Pre-compute the keypoints (we keep the best over all scales, so this has to be done beforehand
817  vector < vector<KeyPoint> > allKeypoints;
818  if( do_keypoints )
819  {
820  // Get keypoints, those will be far enough from the border that no check will be required for the descriptor
821  computeKeyPoints(imagePyramid, maskPyramid, allKeypoints,
822  nfeatures, firstLevel, scaleFactor,
823  edgeThreshold, patchSize, scoreType, fastThreshold);

826  /*vector<KeyPoint> temp;
827 
828  for (int level = 0; level < n_levels; ++level)
829  {

840  }
841  else
842  {

844  KeyPointsFilter::runByImageBorder(_keypoints, image.size(), edgeThreshold);
845 
846  // Cluster the input keypoints depending on the level they were computed at
847  allKeypoints.resize(nlevels);
848  for (vector<KeyPoint>::iterator keypoint = _keypoints.begin(),
849  keypointEnd = _keypoints.end(); keypoint != keypointEnd; ++keypoint)
850  allKeypoints[keypoint->octave].push_back(*keypoint);

861  keypointEnd = keypoints.end(); keypoint != keypointEnd; ++keypoint)
862  keypoint->pt *= scale;
863  }
864  }
865 
866  Mat descriptors;
867  vector<Point> pattern;
868 
869  if( do_descriptors )
870  {
871  int nkeypoints = 0;
872  for (int level = 0; level < nlevels; ++level)
873  nkeypoints += (int)allKeypoints[level].size();

875  _descriptors.release();

904  int offset = 0;
905  for (int level = 0; level < nlevels; ++level)

910

924  GaussianBlur(workingMat, workingMat, Size(7, 7), 2, 2, BORDER_REFLECT_101);
925  computeDescriptors(workingMat, keypoints, desc, pattern, descriptorSize(), WTA_K);
926  }
927 
928  // Copy to the output data
929  if (level != firstLevel)
930  {
931  float scale = getScale(level, firstLevel, scaleFactor);
932  for (vector<KeyPoint>::iterator keypoint = keypoints.begin(),
933  keypointEnd = keypoints.end(); keypoint != keypointEnd; ++keypoint)
934  keypoint->pt *= scale;
