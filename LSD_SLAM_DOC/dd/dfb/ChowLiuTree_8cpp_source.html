177 vector<int> ChowLiuTree::extractChildren(list<info> &remaining_edges, int q) {
178 
179  vector<int> children;

181 
182  while(edge != remaining_edges.end()) {
183  if(edge->word1 == q) {

204  double accumulation = 0;
205 
206  double P00 = JP(word1, false, word2, false);
207  if(P00) accumulation += P00 * log(P00 / (P(word1, false)*P(word2, false)));
208

210  if(P01) accumulation += P01 * log(P01 / (P(word1, false)*P(word2, true)));

214

216  if(P11) accumulation += P11 * log(P11 / (P(word1, true)*P(word2, true)));

239 
240  map<int, int> groups; map<int, int>::iterator groupIt;
