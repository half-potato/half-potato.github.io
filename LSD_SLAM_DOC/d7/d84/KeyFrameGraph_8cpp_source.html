340 void KeyFrameGraph::calculateGraphDistancesToFrame(Frame* startFrame, std::unordered_map< Frame*, int >* distanceMap)
341 {
342  distanceMap->insert(std::make_pair(startFrame, 0));
343 
344  std::multimap< int, Frame* > priorityQueue;
345  priorityQueue.insert(std::make_pair(0, startFrame));
346  while (! priorityQueue.empty())
347  {
348  auto it = priorityQueue.begin();
349  int length = it->first;
350  Frame* frame = it->second;
351  priorityQueue.erase(it);
352 
353  auto mapEntry = distanceMap->find(frame);
354 
355  if (mapEntry != distanceMap->end() && length > mapEntry->second)
356  {
357  continue;
358  }
359 
360  for (Frame* neighbor : frame->neighbors)
361  {
362  auto neighborMapEntry = distanceMap->find(neighbor);
363 
364  if (neighborMapEntry != distanceMap->end() && length + 1 >= neighborMapEntry->second)
365  continue;
366 
367  if (neighborMapEntry != distanceMap->end())
368  neighborMapEntry->second = length + 1;
369  else
370  distanceMap->insert(std::make_pair(neighbor, length + 1));
371  priorityQueue.insert(std::make_pair(length + 1, neighbor));
372  }
373  }
374 }
