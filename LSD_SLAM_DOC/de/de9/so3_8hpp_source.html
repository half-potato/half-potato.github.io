342  inline static
343  const SO3Group<Scalar> expAndTheta(const Tangent & omega,
344  Scalar * theta) {
345  const Scalar theta_sq = omega.squaredNorm();
346  *theta = std::sqrt(theta_sq);
347  const Scalar half_theta = static_cast<Scalar>(0.5)*(*theta);
348 
349  Scalar imag_factor;
350  Scalar real_factor;;
351  if((*theta)<SophusConstants<Scalar>::epsilon()) {
352  const Scalar theta_po4 = theta_sq*theta_sq;
353  imag_factor = static_cast<Scalar>(0.5)
354  - static_cast<Scalar>(1.0/48.0)*theta_sq
355  + static_cast<Scalar>(1.0/3840.0)*theta_po4;
356  real_factor = static_cast<Scalar>(1)
357  - static_cast<Scalar>(0.5)*theta_sq +
358  static_cast<Scalar>(1.0/384.0)*theta_po4;
359  } else {
360  const Scalar sin_half_theta = std::sin(half_theta);
361  imag_factor = sin_half_theta/(*theta);
362  real_factor = std::cos(half_theta);
363  }
364 
365  return SO3Group<Scalar>(Quaternion<Scalar>(real_factor,
366  imag_factor*omega.x(),
367  imag_factor*omega.y(),
368  imag_factor*omega.z()));
369  }

490  inline static
491  const Tangent logAndTheta(const SO3Group<Scalar> & other,
492  Scalar * theta) {
493  const Scalar squared_n
494  = other.unit_quaternion().vec().squaredNorm();
495  const Scalar n = std::sqrt(squared_n);
496  const Scalar w = other.unit_quaternion().w();
497 
498  Scalar two_atan_nbyw_by_n;
499 
500  // Atan-based log thanks to
501  //
502  // C. Hertzberg et al.:
503  // "Integrating Generic Sensor Fusion Algorithms with Sound State
504  // Representation through Encapsulation of Manifolds"
505  // Information Fusion, 2011
506 
507  if (n < SophusConstants<Scalar>::epsilon()) {
508  // If quaternion is normalized and n=0, then w should be 1;
509  // w=0 should never happen here!
510  if (std::abs(w) < SophusConstants<Scalar>::epsilon()) {
511  throw SophusException("Quaternion is not normalized!");
512  }
513  const Scalar squared_w = w*w;
514  two_atan_nbyw_by_n = static_cast<Scalar>(2) / w
515  - static_cast<Scalar>(2)*(squared_n)/(w*squared_w);
516  } else {
517  if (std::abs(w)<SophusConstants<Scalar>::epsilon()) {
518  if (w > static_cast<Scalar>(0)) {
519  two_atan_nbyw_by_n = M_PI/n;
520  } else {
521  two_atan_nbyw_by_n = -M_PI/n;
522  }
523  }else{
524  two_atan_nbyw_by_n = static_cast<Scalar>(2) * atan(n/w) / n;
525  }
526  }
527 
528  *theta = two_atan_nbyw_by_n*n;
529 
530  return two_atan_nbyw_by_n * other.unit_quaternion().vec();
531  }
