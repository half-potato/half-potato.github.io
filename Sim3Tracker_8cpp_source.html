643  //float g0 = (tx * pz - tz * px) / (pz*pz*d);

651  g1s = _mm_mul_ps(g1s,pz2ds);
652

654  __m128 g2s = _mm_sub_ps(pzs, tzs);
655  g2s = _mm_mul_ps(g2s,pz2ds);
656 
657  // calc w_p
658  // float drpdd = gx * g0 + gy * g1; // ommitting the minus
659  __m128 drpdds = _mm_add_ps(
660  _mm_mul_ps(g0s, _mm_load_ps(buf_warped_dx+i)),
661  _mm_mul_ps(g1s, _mm_load_ps(buf_warped_dy+i)));
662 
663  //float w_p = 1.0f / (sigma_i2 + s * drpdd * drpdd);
664  __m128 w_ps = _mm_rcp_ps(_mm_add_ps(sigma_i2s,
665  _mm_mul_ps(drpdds,
666  _mm_mul_ps(drpdds,
667  _mm_mul_ps(depthVarFacs,
668  _mm_load_ps(buf_idepthVar+i))))));
669 
670

672  __m128 w_ds = _mm_rcp_ps(_mm_add_ps(_mm_load_ps(buf_warped_idepthVar+i),
673  _mm_mul_ps(g2s,
674  _mm_mul_ps(g2s,
675  _mm_mul_ps(depthVarFacs,
676  _mm_load_ps(buf_idepthVar+i))))));
677 
678  //float weighted_rp = fabs(rp*sqrtf(w_p));
679  __m128 weighted_rps = _mm_mul_ps(_mm_load_ps(buf_warped_residual+i),
680  _mm_sqrt_ps(w_ps));
681  weighted_rps = _mm_max_ps(weighted_rps, _mm_sub_ps(zeros,weighted_rps));
682 
683  //float weighted_rd = fabs(rd*sqrtf(w_d));
684  __m128 weighted_rds = _mm_mul_ps(_mm_load_ps(buf_residual_d+i),
685  _mm_sqrt_ps(w_ds));
686  weighted_rds = _mm_max_ps(weighted_rds, _mm_sub_ps(zeros,weighted_rds));
687 
688  // depthValid = sv > 0
689  __m128 depthValid = _mm_cmplt_ps(zeros, _mm_load_ps(buf_warped_idepthVar+i)); // bitmask 0xFFFFFFFF for idepth valid, 0x000000 otherwise
690 
691 
692  // float weighted_abs_res = sv > 0 ? weighted_rd+weighted_rp : weighted_rp;
693  __m128 weighted_abs_ress = _mm_add_ps(_mm_and_ps(weighted_rds,depthValid), weighted_rps);
694 
695  //float wh = fabs(weighted_abs_res < huber_res ? 1 : huber_res / weighted_abs_res);
696  __m128 whs = _mm_cmplt_ps(weighted_abs_ress, huber_ress); // bitmask 0xFFFFFFFF for 1, 0x000000 for huber_res_ponly / weighted_rp
697  whs = _mm_or_ps(
698  _mm_and_ps(whs, ones),
699  _mm_andnot_ps(whs, _mm_mul_ps(huber_ress, _mm_rcp_ps(weighted_abs_ress))));
700 
701 
702  if(i+3 < buf_warped_size)
703  {
704  //if(sv > 0) sumRes.numTermsD++;
705  numTermsD = _mm_add_ps(numTermsD,
706  _mm_and_ps(depthValid, ones));
707 
708  //if(sv > 0) sumRes.sumResD += wh * w_d * rd*rd;
709  sumResD = _mm_add_ps(sumResD,
710  _mm_and_ps(depthValid, _mm_mul_ps(whs, _mm_mul_ps(weighted_rds, weighted_rds))));
711 
712  // sumRes.sumResP += wh * w_p * rp*rp;
713  sumResP = _mm_add_ps(sumResP,
714  _mm_mul_ps(whs, _mm_mul_ps(weighted_rps, weighted_rps)));
715  }
716 
717  //*(buf_weight_p+i) = wh * w_p;
718  _mm_store_ps(buf_weight_p+i, _mm_mul_ps(whs, w_ps) );
719 
720  //if(sv > 0) *(buf_weight_d+i) = wh * w_d; else *(buf_weight_d+i) = 0;
721  _mm_store_ps(buf_weight_d+i, _mm_and_ps(depthValid, _mm_mul_ps(whs, w_ds)) );
722 
723  }

729

731  sumRes.meanD = (sumRes.sumResD) / (sumRes.numTermsD);
732  sumRes.meanP = (sumRes.sumResP) / (sumRes.numTermsP);
733 
734  return sumRes;

748 Sim3ResidualStruct Sim3Tracker::calcSim3WeightsAndResidual(
749  const Sim3& referenceToFrame)
750 {
751  float tx = referenceToFrame.translation()[0];
752  float ty = referenceToFrame.translation()[1];
753  float tz = referenceToFrame.translation()[2];
754 
755  Sim3ResidualStruct sumRes;
756  memset(&sumRes, 0, sizeof(Sim3ResidualStruct));
757 
758 
759  float sum_rd=0, sum_rp=0, sum_wrd=0, sum_wrp=0, sum_wp=0, sum_wd=0, sum_num_d=0, sum_num_p=0;
760 
761  for(int i=0;i<buf_warped_size;i++)
762  {
763  float px = *(buf_warped_x+i); // x'
764  float py = *(buf_warped_y+i); // y'
765  float pz = *(buf_warped_z+i); // z'
766 
767  float d = *(buf_d+i); // d
768 
769  float rp = *(buf_warped_residual+i); // r_p
770  float rd = *(buf_residual_d+i); // r_d
771 
772  float gx = *(buf_warped_dx+i); // \delta_x I
773  float gy = *(buf_warped_dy+i); // \delta_y I
774 
775  float s = settings.var_weight * *(buf_idepthVar+i); // \sigma_d^2
776  float sv = settings.var_weight * *(buf_warped_idepthVar+i); // \sigma_d^2'
777 
778 
779  // calc dw/dd (first 2 components):
780  float g0 = (tx * pz - tz * px) / (pz*pz*d);
781  float g1 = (ty * pz - tz * py) / (pz*pz*d);
782  float g2 = (pz - tz) / (pz*pz*d);
783 
784  // calc w_p
785  float drpdd = gx * g0 + gy * g1; // ommitting the minus
786  float w_p = 1.0f / (cameraPixelNoise2 + s * drpdd * drpdd);
787 
788  float w_d = 1.0f / (sv + g2*g2*s);
789 
790  float weighted_rd = fabs(rd*sqrtf(w_d));
791  float weighted_rp = fabs(rp*sqrtf(w_p));
792 
793 
794  float weighted_abs_res = sv > 0 ? weighted_rd+weighted_rp : weighted_rp;
795  float wh = fabs(weighted_abs_res < settings.huber_d ? 1 : settings.huber_d / weighted_abs_res);
796 
797  if(sv > 0)
798  {
799  sumRes.sumResD += wh * w_d * rd*rd;
800  sumRes.numTermsD++;
801  }
802 
803  sumRes.sumResP += wh * w_p * rp*rp;
804  sumRes.numTermsP++;
805 
806 
807  if(plotSim3TrackingIterationInfo)
808  {
809  // for debug
810  *(buf_weight_Huber+i) = wh;
811  *(buf_weight_VarP+i) = w_p;
812  *(buf_weight_VarD+i) = sv > 0 ? w_d : 0;
813 
814 
815  sum_rp += fabs(rp);
816  sum_wrp += fabs(weighted_rp);
817  sum_wp += sqrtf(w_p);
818  sum_num_p++;
819 
820  if(sv > 0)
821  {
822  sum_rd += fabs(weighted_rd);
823  sum_wrd += fabs(rd);
824  sum_wd += sqrtf(w_d);
825  sum_num_d++;
826  }
827  }
828 
829  *(buf_weight_p+i) = wh * w_p;
830 
831  if(sv > 0)
832  *(buf_weight_d+i) = wh * w_d;
833  else
834  *(buf_weight_d+i) = 0;
835 
836  }
837 
838  sumRes.mean = (sumRes.sumResD + sumRes.sumResP) / (sumRes.numTermsD + sumRes.numTermsP);
839  sumRes.meanD = (sumRes.sumResD) / (sumRes.numTermsD);
840  sumRes.meanP = (sumRes.sumResP) / (sumRes.numTermsP);
841 
842  if(plotSim3TrackingIterationInfo)
843  {
844  printf("rd %f, rp %f, wrd %f, wrp %f, wd %f, wp %f\n ",
845  sum_rd/sum_num_d,
846  sum_rp/sum_num_p,
847  sum_wrd/sum_num_d,
848  sum_wrp/sum_num_p,
849  sum_wd/sum_num_d,
850  sum_wp/sum_num_p);
851  }
852  return sumRes;
853 }

939  val4 = _mm_sub_ps(zeros,val4); //val4 = -val4.
940  //v[3] = -px * py * z_sqr * gx +

992 void Sim3Tracker::calcSim3LGS(LGS7 &ls7)
993 {
994  LGS4 ls4;
995  LGS6 ls6;
996  ls6.initialize(width*height);
997  ls4.initialize(width*height);
998 
999  for(int i=0;i<buf_warped_size;i++)
1000  {
1001  float px = *(buf_warped_x+i); // x'
1002  float py = *(buf_warped_y+i); // y'
1003  float pz = *(buf_warped_z+i); // z'
1004 
1005  float wp = *(buf_weight_p+i); // wr/wp
1006  float wd = *(buf_weight_d+i); // wr/wd
1007 
1008  float rp = *(buf_warped_residual+i); // r_p
1009  float rd = *(buf_residual_d+i); // r_d
1010 
1011  float gx = *(buf_warped_dx+i); // \delta_x I
1012  float gy = *(buf_warped_dy+i); // \delta_y I
1013 
1014 
1015  float z = 1.0f / pz;
1016  float z_sqr = 1.0f / (pz*pz);
1017  Vector6 v;
1018  Vector4 v4;
1019  v[0] = z*gx + 0;
1020  v[1] = 0 + z*gy;
1021  v[2] = (-px * z_sqr) * gx +
1022  (-py * z_sqr) * gy;
1023  v[3] = (-px * py * z_sqr) * gx +
1024  (-(1.0 + py * py * z_sqr)) * gy;
1025  v[4] = (1.0 + px * px * z_sqr) * gx +
1026  (px * py * z_sqr) * gy;
1027  v[5] = (-py * z) * gx +
1028  (px * z) * gy;
1029 
1030  // new:
1031  v4[0] = z_sqr;
1032  v4[1] = z_sqr * py;
1033  v4[2] = -z_sqr * px;
1034  v4[3] = z;
1035 
1036  ls6.update(v, rp, wp); // Jac = - v
1037  ls4.update(v4, rd, wd); // Jac = v4
1038 
1039  }
1040 
1041  ls4.finishNoDivide();
1042  ls6.finishNoDivide();
1043 
1044 
1045  ls7.initializeFrom(ls6, ls4);
