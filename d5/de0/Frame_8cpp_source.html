690 void Frame::buildMaxGradients(int level)
691 {
692  require(GRADIENTS, level);
693  boost::unique_lock<boost::mutex> lock2(buildMutex);
694 
695  if(data.maxGradientsValid[level]) return;
696 
697  if(enablePrintDebugInfo && printFrameBuildDebugInfo)
698  printf("CREATE AbsGrad lvl %d for frame %d\n", level, id());
699 
700  int width = data.width[level];
701  int height = data.height[level];
702  if (data.maxGradients[level] == 0)
703  data.maxGradients[level] = FrameMemory::getInstance().getFloatBuffer(width * height);
704 
705  float* maxGradTemp = FrameMemory::getInstance().getFloatBuffer(width * height);
706 
707 
708  // 1. write abs gradients in real data.
709  Eigen::Vector4f* gradxyii_pt = data.gradients[level] + width;
710  float* maxgrad_pt = data.maxGradients[level] + width;
711  float* maxgrad_pt_max = data.maxGradients[level] + width*(height-1);
712 
713  for(; maxgrad_pt < maxgrad_pt_max; maxgrad_pt++, gradxyii_pt++)
714  {
715  float dx = *((float*)gradxyii_pt);
716  float dy = *(1+(float*)gradxyii_pt);
717  *maxgrad_pt = sqrtf(dx*dx + dy*dy);
718  }
719 
720  // 2. smear up/down direction into temp buffer
721  maxgrad_pt = data.maxGradients[level] + width+1;
722  maxgrad_pt_max = data.maxGradients[level] + width*(height-1)-1;
723  float* maxgrad_t_pt = maxGradTemp + width+1;
724  for(;maxgrad_pt<maxgrad_pt_max; maxgrad_pt++, maxgrad_t_pt++)
725  {
726  float g1 = maxgrad_pt[-width];
727  float g2 = maxgrad_pt[0];
728  if(g1 < g2) g1 = g2;
729  float g3 = maxgrad_pt[width];
730  if(g1 < g3)
731  *maxgrad_t_pt = g3;
732  else
733  *maxgrad_t_pt = g1;
734  }
735 
736  float numMappablePixels = 0;
737  // 2. smear left/right direction into real data
738  maxgrad_pt = data.maxGradients[level] + width+1;
739  maxgrad_pt_max = data.maxGradients[level] + width*(height-1)-1;
740  maxgrad_t_pt = maxGradTemp + width+1;
741  for(;maxgrad_pt<maxgrad_pt_max; maxgrad_pt++, maxgrad_t_pt++)
742  {
743  float g1 = maxgrad_t_pt[-1];
744  float g2 = maxgrad_t_pt[0];
745  if(g1 < g2) g1 = g2;
746  float g3 = maxgrad_t_pt[1];
747  if(g1 < g3)
748  {
749  *maxgrad_pt = g3;
750  if(g3 >= MIN_ABS_GRAD_CREATE)
751  numMappablePixels++;
752  }
753  else
754  {
755  *maxgrad_pt = g1;
756  if(g1 >= MIN_ABS_GRAD_CREATE)
757  numMappablePixels++;
758  }
759  }
760 
761  if(level==0)
762  this->numMappablePixels = numMappablePixels;
763 
764  FrameMemory::getInstance().returnBuffer(maxGradTemp);
765 
766  data.maxGradientsValid[level] = true;
767 }

806 
807  const float* idepthSource = data.idepth[level - 1];
808  const float* idepthVarSource = data.idepthVar[level - 1];
809  float* idepthDest = data.idepth[level];
810  float* idepthVarDest = data.idepthVar[level];
811 
812  for(int y=0;y<height;y++)
813  {
814  for(int x=0;x<width;x++)
815  {
816  int idx = 2*(x+y*sw);
817  int idxDest = (x+y*width);
818 
819  float idepthSumsSum = 0;
820  float ivarSumsSum = 0;
821  int num=0;
822 
823  // build sums
824  float ivar;
825  float var = idepthVarSource[idx];
826  if(var > 0)
827  {
828  ivar = 1.0f / var;
829  ivarSumsSum += ivar;

832  }
833

836  {

845  {

849  num++;
850  }

853  if(var > 0)

859  }
