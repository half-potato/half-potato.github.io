229 double FabMap::getNewPlaceLikelihood(const Mat& queryImgDescriptor) {
230  if (flags & MEAN_FIELD) {
231  double logP = 0;
232  bool zq, zpq;
233  if(flags & NAIVE_BAYES) {
234  for (int q = 0; q < clTree.cols; q++) {
235  zq = queryImgDescriptor.at<float>(0,q) > 0;
236 
237  logP += log(Pzq(q, false) * PzqGeq(zq, false) +
238  Pzq(q, true) * PzqGeq(zq, true));
239  }
240  } else {
241  for (int q = 0; q < clTree.cols; q++) {
242  zq = queryImgDescriptor.at<float>(0,q) > 0;
243  zpq = queryImgDescriptor.at<float>(0,pq(q)) > 0;
244 
245  double alpha, beta, p;
246  alpha = Pzq(q, zq) * PzqGeq(!zq, false) * PzqGzpq(q, !zq, zpq);
247  beta = Pzq(q, !zq) * PzqGeq(zq, false) * PzqGzpq(q, zq, zpq);
248  p = Pzq(q, false) * beta / (alpha + beta);
249 
250  alpha = Pzq(q, zq) * PzqGeq(!zq, true) * PzqGzpq(q, !zq, zpq);
251  beta = Pzq(q, !zq) * PzqGeq(zq, true) * PzqGzpq(q, zq, zpq);
252  p += Pzq(q, true) * beta / (alpha + beta);
253 
254  logP += log(p);
255  }
256  }
257  return logP;
258  }
259 
260  if (flags & SAMPLED) {
261  CV_Assert(!trainingImgDescriptors.empty());
262  CV_Assert(numSamples > 0);
263 
264  vector<Mat> sampledImgDescriptors;
265 
266  // TODO: this method can result in the same sample being added
267  // multiple times. Is this desired?
268 
269  for (int i = 0; i < numSamples; i++) {
270  int index = rand() % trainingImgDescriptors.size();
271  sampledImgDescriptors.push_back(trainingImgDescriptors[index]);
272  }
273 
274  vector<IMatch> matches;
275  getLikelihoods(queryImgDescriptor,sampledImgDescriptors,matches);
276 
277  double averageLogLikelihood = -DBL_MAX + matches.front().likelihood + 1;
278  for (int i = 0; i < numSamples; i++) {
279  averageLogLikelihood =
280  logsumexp(matches[i].likelihood, averageLogLikelihood);
281  }
282 
283  return averageLogLikelihood - log((double)numSamples);
284  }
285  return 0;
286 }

288 void FabMap::normaliseDistribution(vector<IMatch>& matches) {
289  CV_Assert(!matches.empty());
290 
291  if (flags & MOTION_MODEL) {
292 
293  matches[0].match = matches[0].likelihood + log(Pnew);
294 
295  if (priorMatches.size() > 2) {
296  matches[1].match = matches[1].likelihood;
297  matches[1].match += log(
298  (2 * (1-mBias) * priorMatches[1].match +
299  priorMatches[1].match +
300  2 * mBias * priorMatches[2].match) / 3);
301  for (size_t i = 2; i < priorMatches.size()-1; i++) {
302  matches[i].match = matches[i].likelihood;
303  matches[i].match += log(
304  (2 * (1-mBias) * priorMatches[i-1].match +
305  priorMatches[i].match +
306  2 * mBias * priorMatches[i+1].match)/3);
307  }
308  matches[priorMatches.size()-1].match =
309  matches[priorMatches.size()-1].likelihood;
310  matches[priorMatches.size()-1].match += log(
311  (2 * (1-mBias) * priorMatches[priorMatches.size()-2].match +
312  priorMatches[priorMatches.size()-1].match +
313  2 * mBias * priorMatches[priorMatches.size()-1].match)/3);
314 
315  for(size_t i = priorMatches.size(); i < matches.size(); i++) {
316  matches[i].match = matches[i].likelihood;
317  }
318  } else {
319  for(size_t i = 1; i < matches.size(); i++) {
320  matches[i].match = matches[i].likelihood;
321  }
322  }
323 
324  double logsum = -DBL_MAX + matches.front().match + 1;
325 
326  //calculate the normalising constant
327  for (size_t i = 0; i < matches.size(); i++) {
328  logsum = logsumexp(logsum, matches[i].match);
329  }
330 
331  //normalise
332  for (size_t i = 0; i < matches.size(); i++) {
333  matches[i].match = exp(matches[i].match - logsum);
334  }
335 
336  //smooth final probabilities
337  for (size_t i = 0; i < matches.size(); i++) {
338  matches[i].match = sFactor*matches[i].match +
339  (1 - sFactor)/matches.size();
340  }
341 
342  //update our location priors
343  priorMatches = matches;
344 
345  } else {
346 
347  double logsum = -DBL_MAX + matches.front().likelihood + 1;
348 
349  for (size_t i = 0; i < matches.size(); i++) {
350  logsum = logsumexp(logsum, matches[i].likelihood);
351  }
352  for (size_t i = 0; i < matches.size(); i++) {
353  matches[i].match = exp(matches[i].likelihood - logsum);
354  }
355  for (size_t i = 0; i < matches.size(); i++) {
356  matches[i].match = sFactor*matches[i].match +
357  (1 - sFactor)/matches.size();
358  }
359  }
360 }

386 double FabMap::PeqGL(int q, bool Lzq, bool eq) {
387  double alpha, beta;
388  alpha = PzqGeq(Lzq, true) * Pzq(q, true);
389  beta = PzqGeq(Lzq, false) * Pzq(q, false);
390 
391  if (eq) {
392  return alpha / (alpha + beta);
393  } else {
394  return 1 - (alpha / (alpha + beta));

405  double p;
406  double alpha, beta;
407 
408  alpha = Pzq(q, zq) * PzqGeq(!zq, false) * PzqGzpq(q, !zq, zpq);
409  beta = Pzq(q, !zq) * PzqGeq( zq, false) * PzqGzpq(q, zq, zpq);

501 void FabMapFBO::getLikelihoods(const Mat& queryImgDescriptor,
502  const vector<Mat>& testImgDescriptors, vector<IMatch>& matches) {
503 
504  multiset<WordStats> wordData;
505  setWordStatistics(queryImgDescriptor, wordData);
506 
507  vector<int> matchIndices;
508  vector<IMatch> queryMatches;
509 
510  for (size_t i = 0; i < testImgDescriptors.size(); i++) {
511  queryMatches.push_back(IMatch(0,i,0,0));
512  matchIndices.push_back(i);
513  }
514 
515  double currBest;
516  double bailedOut = DBL_MAX;
517 
518  for (multiset<WordStats>::iterator wordIter = wordData.begin();
519  wordIter != wordData.end(); wordIter++) {
520  bool zq = queryImgDescriptor.at<float>(0,wordIter->q) > 0;
521  bool zpq = queryImgDescriptor.at<float>(0,pq(wordIter->q)) > 0;
522 
523  currBest = -DBL_MAX;
524 
525  for (size_t i = 0; i < matchIndices.size(); i++) {
526  bool Lzq =
527  testImgDescriptors[matchIndices[i]].at<float>(0,wordIter->q) > 0;
528  queryMatches[matchIndices[i]].likelihood +=
529  log((this->*PzGL)(wordIter->q,zq,zpq,Lzq));
530  currBest =
531  std::max(queryMatches[matchIndices[i]].likelihood, currBest);
532  }
533 
534  if (matchIndices.size() == 1)
535  continue;
536 
537  double delta = std::max(limitbisection(wordIter->V, wordIter->M),
538  -log(rejectionThreshold));
539 
540  vector<int>::iterator matchIter = matchIndices.begin();
541  while (matchIter != matchIndices.end()) {
542  if (currBest - queryMatches[*matchIter].likelihood > delta) {
543  queryMatches[*matchIter].likelihood = bailedOut;
544  matchIter = matchIndices.erase(matchIter);
545  } else {
546  matchIter++;
547  }
548  }
549  }
550 
551  for (size_t i = 0; i < queryMatches.size(); i++) {
552  if (queryMatches[i].likelihood == bailedOut) {
553  queryMatches[i].likelihood = currBest + log(rejectionThreshold);
554  }
555  }
556  matches.insert(matches.end(), queryMatches.begin(), queryMatches.end());
557 
558 }

591 double FabMapFBO::limitbisection(double v, double m) {
592  double midpoint, left_val, mid_val;
593  double left = 0, right = bisectionStart;
594 
595  left_val = bennettInequality(v, m, left) - PsGd;
596 
597  for(int i = 0; i < bisectionIts; i++) {
598 
599  midpoint = (left + right)*0.5;
600  mid_val = bennettInequality(v, m, midpoint)- PsGd;
601 
602  if(left_val * mid_val > 0) {
603  left = midpoint;
604  left_val = mid_val;
605  } else {
606  right = midpoint;
607  }
608  }
609 
610  return (right + left) * 0.5;
611 }

613 double FabMapFBO::bennettInequality(double v, double m, double delta) {
614  double DMonV = delta * m / v;
615  double f_delta = log(DMonV + sqrt(pow(DMonV, 2.0) + 1));
616  return exp((v / pow(m, 2.0))*(cosh(f_delta) - 1 - DMonV * f_delta));
